**Step 1: Add ClaimsStrip helper function** (insert before the EventRow function, around line 450):

```jsx
// ─── Inline claims strip ──────────────────────────────────────────────────────

const CLAIMS_STRIP_IDS = new Set(['user-token', 'exchanged-token', 'agent-actor-token', 'exchanged-token-fallback']);

function fmtSub(sub) {
  if (!sub) return null;
  const s = String(sub);
  return s.length > 14 ? s.slice(0, 12) + '…' : s;
}
function fmtAud(aud) {
  if (!aud) return null;
  const flat = Array.isArray(aud) ? aud[aud.length - 1] : String(aud);
  return flat.split('/').pop() || flat;
}
function fmtScope(scope) {
  if (!scope) return null;
  const s = String(scope);
  return s.length > 60 ? s.slice(0, 58) + '…' : s;
}
function fmtExpiry(exp) {
  if (!exp) return null;
  const secsLeft = Math.round(exp - Date.now() / 1000);
  if (secsLeft < 0) return `expired ${Math.abs(secsLeft)}s ago`;
  if (secsLeft < 60) return `${secsLeft}s`;
  if (secsLeft < 3600) return `${Math.round(secsLeft / 60)}m`;
  return `${Math.round(secsLeft / 3600)}h`;
}
function fmtAct(act) {
  if (!act) return null;
  if (typeof act === 'object') {
    if (act.client_id) return act.client_id;
    // nested act (2-exchange): act.sub or act.client_id at top level is the current actor
    if (act.sub) return `sub:${act.sub}`;
    return JSON.stringify(act).slice(0, 40);
  }
  return String(act).slice(0, 40);
}

/**
 * Compact inline strip showing key claims without opening the inspector.
 * Renders on user-token and exchanged-token events only.
 */
function ClaimsStrip({ event }) {
  if (!CLAIMS_STRIP_IDS.has(event.id)) return null;
  const c = event.claims;
  if (!c) return null;

  const sub     = fmtSub(c.sub);
  const act     = fmtAct(c.act);
  const mayAct  = c.may_act?.client_id ? String(c.may_act.client_id) : null;
  const aud     = fmtAud(c.aud);
  const scope   = fmtScope(c.scope);
  const expiry  = fmtExpiry(c.exp);

  const rows = [
    sub    && { key: 'sub',      val: sub,    cls: '' },
    act    && { key: 'act',      val: act,    cls: 'tcd-cs-act' },
    mayAct && { key: 'may_act',  val: mayAct, cls: 'tcd-cs-may' },
    aud    && { key: 'aud',      val: aud,    cls: 'tcd-cs-aud' },
    scope  && { key: 'scope',    val: scope,  cls: 'tcd-cs-scope' },
    expiry && { key: 'exp',      val: expiry, cls: expiry.includes('ago') ? 'tcd-cs-expired' : '' },
  ].filter(Boolean);

  if (rows.length === 0) return null;

  return (
    <div className="tcd-claims-strip">
      {rows.map(r => (
        <span key={r.key} className={`tcd-cs-item ${r.cls}`}>
          <span className="tcd-cs-key">{r.key}</span>
          <span className="tcd-cs-val">{r.val}</span>
        </span>
      ))}
    </div>
  );
}
```

```jsx
// ─── Exchange mode banner ─────────────────────────────────────────────────────

const EXCHANGE_MODE_MAP = {
  '2-exchange':   { label: '2-Exchange Delegation', cls: 'tcd-exc-banner--teal',  desc: 'Nested act: subject → agent → MCP' },
  'with-actor':   { label: '1-Exchange + actor_token', cls: 'tcd-exc-banner--blue', desc: 'act claim present — BFF delegated' },
  'subject-only': { label: '1-Exchange (no actor)',  cls: 'tcd-exc-banner--slate', desc: 'No act claim — subject-only RFC 8693' },
};

function ExchangeModeBanner({ events }) {
  if (!events || events.length === 0) return null;
  const exchangedEv = events.find(e => e.id === 'exchanged-token' && e.exchangeMethod);
  if (!exchangedEv) return null;
  const info = EXCHANGE_MODE_MAP[exchangedEv.exchangeMethod];
  if (!info) return null;
  return (
    <div className={`tcd-exc-banner ${info.cls}`}>
      <span className="tcd-exc-badge">{info.label}</span>
      <span className="tcd-exc-desc">{info.desc}</span>
    </div>
  );
}
```
