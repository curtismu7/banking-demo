Create banking_api_server/src/__tests__/server-production-guard.test.js.

Use child_process.spawnSync to launch server.js as a subprocess with controlled env vars.
This avoids all the complexity of mocking express-session, Redis, OAuth config, etc.
The process.exit(1) guard fires synchronously before any async initialization, so
the process returns quickly with status 1.

Test 1 — Guard triggers (STAB-03 core requirement):
```javascript
it('exits 1 when SKIP_TOKEN_SIGNATURE_VALIDATION=true in production', () => {
  const result = spawnSync(
    process.execPath,
    [path.join(__dirname, '../../server.js')],
    {
      env: {
        ...process.env,
        SKIP_TOKEN_SIGNATURE_VALIDATION: 'true',
        NODE_ENV: 'production',
        SESSION_SECRET: 'test-session-secret-minimum-32-chars',
        // Strip VERCEL in case it is set in the outer environment
        VERCEL: '',
      },
      timeout: 8000,
      encoding: 'utf8',
    }
  );
  expect(result.status).toBe(1);
  expect(result.stderr).toContain('SKIP_TOKEN_SIGNATURE_VALIDATION');
}, 10000);
```

Test 2 — Guard does NOT trigger in development:
```javascript
it('does not exit immediately when NODE_ENV=development (guard disabled)', () => {
  const result = spawnSync(
    process.execPath,
    // Use -e so we can require server then immediately exit — avoids waiting for app.listen
    ['-e', `
      process.env.SKIP_TOKEN_SIGNATURE_VALIDATION = 'true';
      process.env.NODE_ENV = 'development';
      process.env.SESSION_SECRET = 'test-session-secret-minimum-32-chars';
      // Require server — if guard fires it will process.exit(1)
      // If we get here and guard is NOT triggered (development mode), exit 0 manually
      try { require(${JSON.stringify(path.join(__dirname, '../../server.js'))}); } catch (e) {}
      // Give event loop a tick then force clean exit
      setTimeout(() => process.exit(0), 100);
    `],
    { timeout: 5000, encoding: 'utf8' }
  );
  // Should NOT have exited with 1 due to the guard
  expect(result.status).not.toBe(1);
}, 8000);
```

Note: requireing server.js in development will attempt to start Express + Redis + etc.
The -e approach with setTimeout(exit(0), 100) ensures we exit cleanly after proving
the guard didn't fire, without waiting for the full server to start.
