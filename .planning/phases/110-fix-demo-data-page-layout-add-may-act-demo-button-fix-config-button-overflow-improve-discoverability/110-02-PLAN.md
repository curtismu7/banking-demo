```js
useEffect(() => {
  const observers = NAV_SECTIONS.map(({ id }) => {
    const el = document.getElementById(id);
    if (!el) return null;
    const obs = new IntersectionObserver(
      ([entry]) => { if (entry.isIntersecting) setActiveNav(id); },
      { rootMargin: '-20% 0px -70% 0px', threshold: 0 }
    );
    obs.observe(el);
    return obs;
  });
  return () => observers.forEach(o => o?.disconnect());
}, []);
```

Identify the existing config store pattern (e.g., `configStore.get(key)` / `configStore.set(key, value)` or `getConfig(key)` / `setConfig(key, value)`). Use the same pattern.

Add to the appropriate route file (likely `demoRoute.js` or a new `configRoute.js` if one exists):

```js
// GET /api/config/token-endpoint-auth
router.get('/token-endpoint-auth', requireBffSession, async (req, res) => {
  try {
    const ai = await configStore.get('ai_agent_token_endpoint_auth_method') || '';
    const mcp = await configStore.get('mcp_exchanger_token_endpoint_auth_method') || '';
    res.json({
      ai_agent_token_endpoint_auth_method: ai,
      mcp_exchanger_token_endpoint_auth_method: mcp,
    });
  } catch (err) {
    res.status(500).json({ error: 'Failed to read token endpoint auth config' });
  }
});

// PATCH /api/config/token-endpoint-auth
router.patch('/token-endpoint-auth', requireBffSession, async (req, res) => {
  const VALID = ['client_secret_basic', 'client_secret_post', 'client_secret_jwt', null, ''];
  const { ai_agent_token_endpoint_auth_method: ai, mcp_exchanger_token_endpoint_auth_method: mcp } = req.body;
  if (!VALID.includes(ai) || !VALID.includes(mcp)) {
    return res.status(400).json({ error: 'Invalid auth method value' });
  }
  try {
    if (ai !== undefined) {
      if (ai) await configStore.set('ai_agent_token_endpoint_auth_method', ai);
      else await configStore.delete('ai_agent_token_endpoint_auth_method').catch(() => {});
    }
    if (mcp !== undefined) {
      if (mcp) await configStore.set('mcp_exchanger_token_endpoint_auth_method', mcp);
      else await configStore.delete('mcp_exchanger_token_endpoint_auth_method').catch(() => {});
    }
    res.json({ ok: true });
  } catch (err) {
    res.status(500).json({ error: 'Failed to save token endpoint auth config' });
  }
});
```

Determine prefix by reading how routes are mounted in `banking_api_server/app.js` or `server.js`. Keep consistent with existing patterns.

**Token exchange read-back:** In the token exchange service (either `agentMcpTokenService.js` or `oauthService.js`), where `AI_AGENT_TOKEN_ENDPOINT_AUTH_METHOD` env var is read, also check the config store key as an override:
