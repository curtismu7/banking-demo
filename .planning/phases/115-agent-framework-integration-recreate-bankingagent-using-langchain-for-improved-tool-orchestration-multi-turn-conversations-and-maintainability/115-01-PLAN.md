Create a utility file that defines LangChain `Tool` objects for each MCP tool. Each LangChain Tool wraps the MCP tool and handles parameter marshaling.

**File structure:**
```javascript
// banking_api_server/utils/mcpToolRegistry.js

import { Tool } from '@langchain/core/tools';
import { z } from 'zod';
import { callMcpTool } from '../services/bankingAgentService'; // Existing service

/**
 * Wrap an MCP tool as a LangChain Tool
 * Handles parameter schema, description, and invocation
 */
class McpToolWrapper extends Tool {
  constructor(mcpToolName, description, schema) {
    super();
    this.name = mcpToolName;
    this.description = description;
    this.schema = schema;
  }

  async _call(input) {
    // LangChain calls this method when tool is invoked
    // 'input' is a string or structured object (depending on schema)
    try {
      const result = await callMcpTool(this.name, input);
      return JSON.stringify(result);
    } catch (error) {
      return JSON.stringify({ error: error.message });
    }
  }
}

/**
 * Initialize tool registry with all available MCP tools
 * Return array of LangChain Tools
 */
export function createMcpToolRegistry() {
  return [
    new McpToolWrapper(
      'get_my_accounts',
      'Retrieve list of all user accounts with balances',
      z.object({})  // No parameters
    ),
    new McpToolWrapper(
      'create_transfer',
      'Transfer money from one account to another',
      z.object({
        from_account_id: z.string(),
        to_account_id: z.string(),
        amount: z.number().positive(),
        description: z.string().optional(),
      })
    ),
    new McpToolWrapper(
      'create_deposit',
      'Deposit funds into an account',
      z.object({
        account_id: z.string(),
        amount: z.number().positive(),
        description: z.string().optional(),
      })
    ),
    new McpToolWrapper(
      'create_withdrawal',
      'Withdraw funds from an account',
      z.object({
        account_id: z.string(),
        amount: z.number().positive(),
        description: z.string().optional(),
      })
    ),
    // Add more tools as discovered from /api/mcp/inspector/tools
  ];
}

export { McpToolWrapper };
```

**Key requirements:**
- Each tool has a name, description (used by LangChain for routing), and schema (param validation)
- `_call()` method invokes the underlying MCP tool via existing `callMcpTool()`
- Schema validation using Zod (errors caught and returned as JSON)
- Results returned as JSON strings (LangChain standard)

**Notes:**
- Parameter schema should match MCP tool signature
- Use Zod for runtime validation (helps catch invalid params before calling MCP)
- Error handling: return structured error JSON so agent can retry or ask user

Create main LangChain agent service that coordinates agent loop, memory, and tool execution.

**File structure:**
```javascript
// banking_api_server/services/bankingAgentLangChainService.js

import { ChatAnthropic } from '@langchain/anthropic';
import { AgentExecutor, createOpenAIFunctionsAgent } from 'langchain/agents';
import { ConversationBufferMemory } from 'langchain/memory';
import { createMcpToolRegistry } from '../utils/mcpToolRegistry';

/**
 * LangChain Banking Agent Service
 * Coordinates agent executor, memory, and tool calls
 */

/**
 * Initialize the banking agent with LLM, tools, and memory
 */
export async function initializeBankingAgent() {
  // Initialize Claude 3 LLM
  const llm = new ChatAnthropic({
    modelName: 'claude-3-opus-20240229',  // or claude-3-sonnet, claude-3-haiku
    temperature: 0.7,
    maxTokens: 1024,
    apiKey: process.env.ANTHROPIC_API_KEY,  // Must be set in .env or Vercel secrets
  });

  // Create MCP tool registry
  const tools = createMcpToolRegistry();

  // Initialize conversation memory
  const memory = new ConversationBufferMemory({
    memoryKey: 'chat_history',
    returnMessages: true,
  });

  // Create agent (use struct agent for tool calling)
  const agent = await createOpenAIFunctionsAgent({
    llm,
    tools,
    systemPrompt: `You are a banking assistant helping users manage their accounts.
You can help with:
- Viewing account balances
- Transferring money between accounts
- Depositing funds
- Withdrawing funds

Always confirm high-value transactions before executing them.
Be helpful and clear in your responses.`,
  });

  // Create executor
  const executor = new AgentExecutor({
    agent,
    tools,
    memory,
    verbose: false,  // Set true for debugging
  });

  return executor;
}

/**
 * Process a user message through the agent
 * Returns agent response + any token events
 */
export async function processBankingAgentMessage(message, executor, userId) {
  try {
    const response = await executor.invoke({
      input: message,
      userId,  // Pass user context if needed
    });

    return {
      success: true,
      message: response.output,
      // Note: tokenEvents will be added in Plan 02 (auth/session integration)
    };
  } catch (error) {
    console.error('[bankingAgentLangChain] Agent error:', error);
    return {
      success: false,
      error: error.message || 'Agent error',
    };
  }
}
```

**Key requirements:**
- Agent initialized with Claude 3 (Opus for reasoning, Sonnet for cost, Haiku for speed)
- ConversationBufferMemory maintains chat history
- System prompt sets agent's persona and capabilities
- AgentExecutor manages tool invocation loop
- `processBankingAgentMessage()` is the main call interface

**Configuration notes:**
- `ANTHROPIC_API_KEY` must be set in environment (set in Vercel secrets for production)
- Temperature 0.7 balances creativity and consistency
- maxTokens 1024 limits response size (adjust based on needs)
- verbosity: false in production, true during debugging

**Design notes:**
- Agent loop runs inside the BFF (banking_api_server)
- Memory is in-memory (ConversationBufferMemory) — suitable for demo, not production
- Each user session gets its own executor + memory (stateless, no cross-session leakage)
- Token events (RFC 8693 tracking) will be added in Plan 02

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User Input (message) | User messages processed by LLM — validate length, sanitize for prompt injection |
| Tool Parameters | LLM generates tool params — validate with Zod before calling MCP tools |
| Tool Results | MCP results returned to LLM — sanitize for token limit, no sensitive data leaks |
| API Keys | Anthropic API key must not be exposed — set in process.env only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-115-01-01 | Tampering | LLM generates tool params | mitigate | Zod schema validates all params before calling MCP tools; invalid params rejected with error message |
| T-115-01-02 | Elevation of Privilege | User tricks agent to call high-value operations | mitigate | HITL gate preserved (Plan 02 challenge gate); all operations require auth context |
| T-115-01-03 | Information Disclosure | Tool results contain sensitive data | mitigate | Tool results reviewed in Plan 02 auth flow; PII never logged; tokenEvents tracked separately |
| T-115-01-04 | Denial of Service | Large messages cause token explosion | mitigate | maxTokens capped at 1024; message queue timeout; rate limiting at BFF level |

</threat_model>
