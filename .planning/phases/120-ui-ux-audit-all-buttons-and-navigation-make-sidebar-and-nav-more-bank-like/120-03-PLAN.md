       import React, { useState } from 'react';

       export default function TopNav({ user, onLogout }) {
         const [mobileMenuOpen, setMobileMenuOpen] = useState(false);
         const [activeDropdown, setActiveDropdown] = useState(null);
         const navigate = useNavigate();
         const isAdmin = user?.role === 'admin';
         
         // Groups for dropdowns (reuse nav structure from SideNav)
         const navGroups = isAdmin ? ADMIN_GROUPS : USER_GROUPS;
         
         const handleNavClick = (item) => {
           if (item.to) {
             navigate(item.to);
             setMobileMenuOpen(false);  // Close mobile menu after nav
           } else if (item.action) {
             // Handle action-type items (buttons)
           }
         };
         
         return (
           <nav className="topnav">
             {/* Left: Logo/Branding */}
             <div className="topnav-brand">
               <h1>Banking Demo</h1>  {/* Or use logo image */}
             </div>
             
             {/* Center: Grouped Dropdowns (hidden on mobile) */}
             <div className="topnav-groups">
               {navGroups.map((group) => (
                 <NavDropdown
                   key={group.group}
                   label={group.group}
                   items={group.items}
                   onSelect={handleNavClick}
                   isAdmin={isAdmin}
                 />
               ))}
             </div>
             
             {/* Right: Search + User Menu + Hamburger */}
             <div className="topnav-right">
               <SearchBar />
               <UserMenu user={user} onLogout={onLogout} />
               
               {/* Hamburger (mobile only) */}
               <button
                 className="topnav-hamburger"
                 onClick={() => setMobileMenuOpen(!mobileMenuOpen)}
                 aria-label="Toggle menu"
               >
                 <MdMenu size={24} />
               </button>
             </div>
             
             {/* Mobile Menu Drawer */}
             {mobileMenuOpen && (
               <div className="topnav-mobile-menu">
                 {navGroups.map((group) => (
                   <NavDropdown
                     key={group.group}
                     label={group.group}
                     items={group.items}
                     onSelect={handleNavClick}
                     isAdmin={isAdmin}
                     isMobile
                   />
                 ))}
               </div>
             )}
           </nav>
         );
       }
       
       // Sub-component: NavDropdown (with grouping)
       function NavDropdown({ label, items, onSelect, isAdmin, isMobile = false }) {
         const [open, setOpen] = useState(false);
         
         return (
           <div className={`nav-dropdown ${open ? 'open' : ''}`} onMouseLeave={() => !isMobile && setOpen(false)}>
             <button
               className="nav-dropdown-toggle"
               onClick={() => setOpen(!open)}
               aria-expanded={open}
             >
               {label}
               <span className="dropdown-arrow">▼</span>
             </button>
             
             {open && (
               <div className="dropdown-menu">
                 {items
                   .filter((item) => !item.adminOnly || isAdmin)  // Filter admin-only items
                   .map((item, idx) => (
                     <button
                       key={idx}
                       className="dropdown-item"
                       onClick={() => {
                         onSelect(item);
                         setOpen(false);
                       }}
                     >
                       {item.label}
                     </button>
                   ))}
               </div>
             )}
           </div>
         );
       }
       
       // Sub-component: SearchBar
       function SearchBar() {
         const [open, setOpen] = useState(false);
         return (
           <div className="search-bar">
             <input
               type="text"
               placeholder="Search..."
               className={`search-input ${open ? 'open' : ''}`}
               onFocus={() => setOpen(true)}
               onBlur={() => setOpen(false)}
             />
           </div>
         );
       }
