Add a new helper method `createSchemaAttribute(attrName, displayName)` at the bottom of the class (before `enableTokenCustomization`):

```javascript
async createSchemaAttribute(attrName, displayName) {
  try {
    // Get the PINGONE_USER schema ID
    const schemasResp = await this.makeRequest('GET', '/schemas');
    const schemas = schemasResp.data._embedded?.schemas || [];
    const userSchema = schemas.find(s => s.schemaType === 'PINGONE_USER' || s.name === 'User');
    if (!userSchema) throw new Error('PINGONE_USER schema not found');

    // Check if attribute already exists
    const attrsResp = await this.makeRequest('GET', `/schemas/${userSchema.id}/attributes`);
    const attrs = attrsResp.data._embedded?.attributes || [];
    if (attrs.find(a => a.name === attrName)) {
      return { success: true, exists: true };
    }

    // Create the attribute
    await this.makeRequest('POST', `/schemas/${userSchema.id}/attributes`, {
      name: attrName,
      displayName: displayName,
      type: 'STRING',
      enabled: true,
      required: false,
      unique: false,
      multivalued: false
    });
    return { success: true, exists: false };
  } catch (error) {
    // Non-fatal — log but don't block provisioning
    return { success: false, error: error.message };
  }
}
```
