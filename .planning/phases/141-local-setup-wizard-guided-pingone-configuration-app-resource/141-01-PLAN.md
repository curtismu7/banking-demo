```javascript
/**
 * Get the default User schema ID for this environment
 */
async getSchemaId() {
  const response = await this.makeRequest('GET', '/v1/environments/' + this.envId + '/schemas');
  const schemas = response.data._embedded?.schemas || [];
  const userSchema = schemas.find(s => s.name === 'User') || schemas[0];
  if (!userSchema) throw new Error('No User schema found in environment');
  return userSchema.id;
}

/**
 * Create a custom attribute on the User schema
 */
async createCustomAttribute(schemaId, name, description = '') {
  try {
    const existing = await this.makeRequest('GET', '/v1/environments/' + this.envId + '/schemas/' + schemaId + '/attributes');
    const attrs = existing.data._embedded?.attributes || [];
    if (attrs.find(a => a.name === name)) {
      return { success: true, existed: true, name };
    }
  } catch (_) { /* probe failed, try create anyway */ }

  const response = await this.makeRequest('POST', '/v1/environments/' + this.envId + '/schemas/' + schemaId + '/attributes', {
    name,
    type: 'STRING',
    enabled: true,
    unique: false,
    multivalued: false,
    description
  });
  return { success: true, existed: false, name, id: response.data.id };
}

/**
 * Add a token claim mapping to an OIDC application
 * @param {string} appId - Application ID
 * @param {string} claimName - Claim name (e.g. bankingPrincipalUserId)
 * @param {string} spelExpression - SPEL value (e.g. ${user.bankingPrincipalUserId})
 */
async addAttributeMapping(appId, claimName, spelExpression) {
  try {
    const existing = await this.makeRequest('GET', '/v1/environments/' + this.envId + '/applications/' + appId + '/attributes');
    const attrs = existing.data._embedded?.attributes || [];
    if (attrs.find(a => a.name === claimName)) {
      return { success: true, existed: true };
    }
  } catch (_) { /* probe failed, try create anyway */ }

  await this.makeRequest('POST', '/v1/environments/' + this.envId + '/applications/' + appId + '/attributes', {
    name: claimName,
    value: spelExpression,
    required: false
  });
  return { success: true, existed: false };
}

/**
 * Retrieve the client secret for an application
 */
async getApplicationSecret(appId) {
  try {
    const response = await this.makeRequest('GET', '/v1/environments/' + this.envId + '/applications/' + appId + '/secret');
    return response.data.secret || null;
  } catch (_) {
    return null; // Secret not retrievable (may require additional permissions)
  }
}

/**
 * Run headless client_credentials smoke test for a single worker app
 */
async runSmokeTest(appName, clientId, clientSecret) {
  try {
    const response = await axios.post(
      'https://auth.pingone.' + this.region + '/' + this.envId + '/as/token',
      'grant_type=client_credentials',
      {
        auth: { username: clientId, password: clientSecret },
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        timeout: 15000
      }
    );
    const hasToken = !!response.data.access_token;
    return { app: appName, success: hasToken, tokenType: response.data.token_type, error: null };
  } catch (error) {
    const msg = error.response?.data?.error_description || error.response?.data?.error || error.message;
    return { app: appName, success: false, error: msg };
  }
}
```

Note: `makeRequest` in `pingoneProvisionService.js` prepends `this.baseURL` which is `https://api.pingone.{region}/{envId}`. The schema/attribute mapping endpoints are at `https://api.pingone.{region}/v1/environments/{envId}/schemas/...` — use the full path in makeRequest by passing the full URL manually via axios for schema endpoints, OR adjust `makeRequest` to accept a full URL override. Use a separate `makeAbsoluteRequest` helper:

```javascript
async makeAbsoluteRequest(method, fullUrl, data = null) {
  const headers = {
    'Authorization': 'Bearer ' + this.workerToken,
    'Content-Type': 'application/json'
  };
  try {
    const config = { method, url: fullUrl, headers };
    if (data) config.data = data;
    return await axios(config);
  } catch (error) {
    const msg = error.response?.data?.details?.[0]?.message || error.response?.data?.message || error.message;
    throw new Error(method + ' ' + fullUrl + ' failed: ' + msg);
  }
}
```

Then in the schema methods above, replace `this.makeRequest` calls with `this.makeAbsoluteRequest` using:

```javascript
/**
 * PUBLIC wizard router — no authentication required.
 * Rate limited. Accepts worker credentials in request body.
 * Mirrors /api/admin/setup/* but accessible pre-auth for first-time setup.
 */
const publicRouter = express.Router();

const wizardRateLimit = rateLimit({
  windowMs: 60 * 1000,
  max: 5,
  standardHeaders: true,
  legacyHeaders: false,
  message: { ok: false, error: 'too_many_requests', message: 'Too many setup requests. Wait 1 minute.' }
});

// POST /api/setup/wizard/run — SSE provisioning stream (public)
publicRouter.post('/run', wizardRateLimit, async (req, res) => {
  // Identical body to /api/admin/setup/run but no requireAdmin
  const { envId, workerClientId, workerClientSecret, region, publicAppUrl, stepUpAcrValue } = req.body;
  if (!envId || !workerClientId || !workerClientSecret) {
    return res.status(400).json({ error: 'missing_required_fields', message: 'envId, workerClientId, and workerClientSecret are required' });
  }
  res.writeHead(200, {
    'Content-Type': 'text/event-stream',
    'Cache-Control': 'no-cache',
    'Connection': 'keep-alive',
    'X-Accel-Buffering': 'no'
  });
  const onStep = (step) => { try { res.write('data: ' + JSON.stringify(step) + '\n\n'); } catch (_) {} };
  onStep({ step: 'connected', icon: '🔗', message: 'Connected to setup wizard' });
  try {
    const config = {
      envId, workerClientId, workerClientSecret,
      region: region || 'com',
      publicAppUrl: publicAppUrl || process.env.PUBLIC_APP_URL || 'http://localhost:3001',
      stepUpAcrValue: stepUpAcrValue || 'Multi_Factor',
      isVercel: !!process.env.VERCEL
    };
    const result = await provisionEnvironment(config, onStep);
    onStep({ step: 'complete', icon: '🎉', message: 'Setup complete!', result: result.provisioned });
  } catch (err) {
    onStep({ step: 'error', icon: '❌', message: 'Setup failed: ' + err.message });
  }
  try { res.write('data: [DONE]\n\n'); res.end(); } catch (_) {}
});

// POST /api/setup/wizard/validate — credential validation (public)
publicRouter.post('/validate', wizardRateLimit, async (req, res) => {
  const { envId, workerClientId, workerClientSecret, region } = req.body;
  if (!envId || !workerClientId || !workerClientSecret) {
    return res.status(400).json({ error: 'missing_required_fields', message: 'envId, workerClientId, and workerClientSecret are required' });
  }
  try {
    const { PingOneProvisionService } = require('../services/pingoneProvisionService');
    const svc = new PingOneProvisionService();
    await svc.initialize(envId, workerClientId, workerClientSecret, region || 'com');
    res.json({ valid: true, message: 'Worker credentials valid', populationId: svc.populationId });
  } catch (err) {
    res.status(400).json({ valid: false, message: 'Invalid credentials', error: err.message });
  }
});

// GET /api/setup/wizard/status — discovery probe (public)
publicRouter.get('/status', wizardRateLimit, async (req, res) => {
  const { envId, workerClientId, workerClientSecret, region } = req.query;
  if (!envId || !workerClientId || !workerClientSecret) {
    return res.status(400).json({ error: 'missing_required_fields', message: 'Query params envId, workerClientId, workerClientSecret required' });
  }
  try {
    const { PingOneProvisionService } = require('../services/pingoneProvisionService');
    const svc = new PingOneProvisionService();
    await svc.initialize(envId, workerClientId, workerClientSecret, region || 'com');
    const [resourceServer, adminApp, userApp, agentApp] = await Promise.allSettled([
      svc.findResourceByName('resource', 'Super Banking API Resource'),
      svc.findResourceByName('application', 'Super Banking Admin App'),
      svc.findResourceByName('application', 'Super Banking User App'),
      svc.findResourceByName('application', 'Super Banking Agent App')
    ]);
    res.json({
      ok: true,
      existing: {
        resourceServer: resourceServer.value ?? null,
        adminApp: adminApp.value ?? null,
        userApp: userApp.value ?? null,
        agentApp: agentApp.value ?? null
      }
    });
  } catch (err) {
    res.status(400).json({ ok: false, error: err.message });
  }
});

// POST /api/setup/wizard/smoke-test — headless OAuth test per worker app (public)
publicRouter.post('/smoke-test', wizardRateLimit, async (req, res) => {
  const { envId, workerClientId, workerClientSecret, region, apps } = req.body;
  if (!envId || !workerClientId || !workerClientSecret) {
    return res.status(400).json({ error: 'missing_required_fields' });
  }
  try {
    const { PingOneProvisionService } = require('../services/pingoneProvisionService');
    const svc = new PingOneProvisionService();
    await svc.initialize(envId, workerClientId, workerClientSecret, region || 'com');
    // apps = [{ name, clientId, clientSecret }]
    const testApps = Array.isArray(apps) && apps.length > 0
      ? apps
      : [{ name: 'Worker App', clientId: workerClientId, clientSecret: workerClientSecret }];
    const results = await Promise.all(
      testApps.map(app => svc.runSmokeTest(app.name, app.clientId, app.clientSecret))
    );
    const allPass = results.every(r => r.success);
    res.json({ ok: allPass, results });
  } catch (err) {
    res.status(500).json({ ok: false, error: err.message });
  }
});

// POST /api/setup/wizard/recreate — recreate a specific resource (public)
publicRouter.post('/recreate', wizardRateLimit, async (req, res) => {
  const { resource, envId, workerClientId, workerClientSecret, region } = req.body;
  if (!resource || !envId || !workerClientId || !workerClientSecret) {
    return res.status(400).json({ error: 'missing_required_fields' });
  }
  res.writeHead(200, { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache', 'Connection': 'keep-alive', 'X-Accel-Buffering': 'no' });
  const onStep = (step) => { try { res.write('data: ' + JSON.stringify(step) + '\n\n'); } catch (_) {} };
  try {
    const config = { envId, workerClientId, workerClientSecret, region: region || 'com' };
    const result = await recreateResource(config, resource);
    onStep(result.success
      ? { step: 'recreate-success', icon: '✅', message: result.message }
      : { step: 'recreate-error', icon: '❌', message: result.error });
  } catch (err) {
    onStep({ step: 'recreate-error', icon: '❌', message: 'Recreate failed: ' + err.message });
  }
  try { res.write('data: [DONE]\n\n'); res.end(); } catch (_) {}
});

module.exports = { adminRouter: router, publicRouter };
```
