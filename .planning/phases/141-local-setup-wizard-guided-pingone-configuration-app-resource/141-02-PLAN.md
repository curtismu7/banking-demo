# Plan 141-02: React — SetupWizardPage (5-Section Accordion)

## Goal
Create `SetupWizardPage.js` — a new React page at `/setup/wizard` with a 5-section accordion that walks the user from credential entry through PingOne object creation, env file generation, and smoke testing. Link it from `SetupPage`. No authentication required.

## must_haves
- [ ] `/setup/wizard` route renders `SetupWizardPage` without auth guard
- [ ] 5 accordion sections in order: Credentials → Discovery → Create → Env File → Smoke Test
- [ ] Each section gates on the previous completing successfully
- [ ] Section 1 has form with all required fields; "Validate" button calls `/api/setup/wizard/validate`
- [ ] Section 2 calls `GET /api/setup/wizard/status` and shows existing vs missing objects table
- [ ] Section 3 streams SSE from `POST /api/setup/wizard/run` with per-item progress rows
- [ ] Section 4 shows generated `.env` diff and a "Write to disk" confirmation
- [ ] Section 5 calls `POST /api/setup/wizard/smoke-test` and shows per-app pass/fail
- [ ] `SetupPage.js` has a "Launch Setup Wizard →" button/link
- [ ] No auth guard on `/setup/wizard` route in `App.js`

```jsx
// banking_api_ui/src/components/SetupWizardPage.js
import React, { useState, useRef, useEffect, useCallback } from 'react';
import { Link } from 'react-router-dom';
import apiClient from '../services/apiClient';
import { notifyError } from '../utils/appToast';
import './SetupWizardPage.css';

// Section status values
const STATUS = { ACTIVE: 'active', COMPLETE: 'complete', LOCKED: 'locked' };

const REGION_OPTIONS = [
  { value: 'com', label: 'North America (pingone.com)' },
  { value: 'eu', label: 'Europe (pingone.eu)' },
  { value: 'ca', label: 'Canada (pingone.ca)' },
  { value: 'asia', label: 'Asia (pingone.asia)' },
  { value: 'com.au', label: 'Australia (pingone.com.au)' },
  { value: 'sg', label: 'Singapore (pingone.sg)' },
];

const AUTH_METHOD_OPTIONS = [
  { value: 'client_secret_basic', label: 'Client Secret Basic (recommended)' },
  { value: 'client_secret_post', label: 'Client Secret Post' },
];

function SectionHeader({ title, status, index, onToggle, isOpen }) {
  const icon = status === STATUS.COMPLETE ? '✅' : status === STATUS.LOCKED ? '🔒' : '▶';
  return (
    <button
      className={`swp-section-header swp-section-header--${status}`}
      onClick={onToggle}
      disabled={status === STATUS.LOCKED}
      type="button"
    >
      <span className="swp-section-number">{index}</span>
      <span className="swp-section-icon">{icon}</span>
      <span className="swp-section-title">{title}</span>
      {status !== STATUS.LOCKED && (
        <span className="swp-section-chevron">{isOpen ? '▲' : '▼'}</span>
      )}
    </button>
  );
}

function LogRow({ entry }) {
  const cls = entry.step?.includes('error') || entry.step?.includes('fail')
    ? 'swp-log-row--error'
    : entry.step?.includes('warning') || entry.step?.includes('skip') || entry.step?.includes('exist')
    ? 'swp-log-row--warn'
    : 'swp-log-row--ok';
  return (
    <div className={`swp-log-row ${cls}`}>
      <span className="swp-log-icon">{entry.icon || '•'}</span>
      <span className="swp-log-message">{entry.message}</span>
    </div>
  );
}

export default function SetupWizardPage() {
  // ── Credentials form state ──────────────────────────────────────────────
  const [creds, setCreds] = useState({
    envId: '',
    workerClientId: '',
    workerClientSecret: '',
    region: 'com',
    tokenAuthMethod: 'client_secret_basic',
    publicAppUrl: window.location.origin.replace(':3000', ':3001').replace(':4000', ':3001'),
    stepUpAcrValue: 'Multi_Factor',
  });
  const [showSecret, setShowSecret] = useState(false);
  const [credErrors, setCredErrors] = useState({});
  const [validating, setValidating] = useState(false);

  // ── Discovery state ─────────────────────────────────────────────────────
  const [discovery, setDiscovery] = useState(null);
  const [discovering, setDiscovering] = useState(false);

  // ── Create state ────────────────────────────────────────────────────────
  const [logEntries, setLogEntries] = useState([]);
  const [running, setRunning] = useState(false);
  const [forceRecreate, setForceRecreate] = useState(false);
  const [provisioned, setProvisioned] = useState(null);
  const abortRef = useRef(null);
  const logEndRef = useRef(null);

  // ── Env File state ──────────────────────────────────────────────────────
  const [envContents, setEnvContents] = useState('');
  const [envWritten, setEnvWritten] = useState(false);

  // ── Smoke test state ────────────────────────────────────────────────────
  const [smokeResults, setSmokeResults] = useState(null);
  const [smoking, setSmoking] = useState(false);

  // ── Section open/status state ───────────────────────────────────────────
  const [sections, setSections] = useState({
    credentials: STATUS.ACTIVE,
    discovery: STATUS.LOCKED,
    create: STATUS.LOCKED,
    envFile: STATUS.LOCKED,
    smokeTest: STATUS.LOCKED,
  });
  const [openSection, setOpenSection] = useState('credentials');

  useEffect(() => {
    logEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  });

  const unlockSection = useCallback((prev, done, next) => {
    setSections(s => ({ ...s, [done]: STATUS.COMPLETE, [next]: STATUS.ACTIVE }));
    setOpenSection(next);
  }, []);

  // ── SSE streaming ───────────────────────────────────────────────────────
  const streamSSE = useCallback(async (response) => {
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';
    let lastResult = null;
    try {
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop();
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = line.slice(6);
            if (data === '[DONE]') return lastResult;
            try {
              const step = JSON.parse(data);
              setLogEntries(prev => [...prev, { ...step, timestamp: step.timestamp || new Date().toISOString() }]);
              if (step.step === 'complete' && step.result) lastResult = step.result;
            } catch (_) {}
          }
        }
      }
    } finally {
      reader.releaseLock();
    }
    return lastResult;
  }, []);

  // ── Section 1: Credentials ──────────────────────────────────────────────
  const validateCreds = () => {
    const errs = {};
    if (!creds.envId.trim()) errs.envId = 'Required';
    if (!creds.workerClientId.trim()) errs.workerClientId = 'Required';
    if (!creds.workerClientSecret.trim()) errs.workerClientSecret = 'Required';
    if (!creds.publicAppUrl.trim()) errs.publicAppUrl = 'Required';
    setCredErrors(errs);
    return Object.keys(errs).length === 0;
  };

  const handleValidate = async () => {
    if (!validateCreds()) return;
    setValidating(true);
    try {
      const { data } = await apiClient.post('/api/setup/wizard/validate', {
        envId: creds.envId,
        workerClientId: creds.workerClientId,
        workerClientSecret: creds.workerClientSecret,
        region: creds.region,
      });
      if (data.valid) {
        unlockSection(null, 'credentials', 'discovery');
      } else {
        notifyError(data.message || 'Credentials invalid');
      }
    } catch (err) {
      const msg = err.response?.data?.message || err.message || 'Validation failed';
      notifyError(msg);
    } finally {
      setValidating(false);
    }
  };

  // ── Section 2: Discovery ────────────────────────────────────────────────
  const handleDiscover = async () => {
    setDiscovering(true);
    try {
      const params = new URLSearchParams({
        envId: creds.envId,
        workerClientId: creds.workerClientId,
        workerClientSecret: creds.workerClientSecret,
        region: creds.region,
      });
      const { data } = await apiClient.get('/api/setup/wizard/status?' + params.toString());
      setDiscovery(data.existing || {});
      unlockSection(null, 'discovery', 'create');
    } catch (err) {
      notifyError(err.response?.data?.error || err.message || 'Discovery failed');
    } finally {
      setDiscovering(false);
    }
  };

  // ── Section 3: Create ───────────────────────────────────────────────────
  const handleRun = async () => {
    setRunning(true);
    setLogEntries([]);
    setProvisioned(null);
    abortRef.current = new AbortController();
    try {
      const response = await fetch('/api/setup/wizard/run', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({
          ...creds,
          forceRecreate,
        }),
        signal: abortRef.current.signal,
      });
      if (!response.ok) throw new Error('HTTP ' + response.status);
      const result = await streamSSE(response);
      if (result) {
        setProvisioned(result);
        // Generate env contents
        const envLines = generateEnvContents(result, creds);
        setEnvContents(envLines);
        unlockSection(null, 'create', 'envFile');
      }
    } catch (err) {
      if (err.name !== 'AbortError') {
        setLogEntries(prev => [...prev, { step: 'error', icon: '❌', message: 'Run failed: ' + err.message }]);
      }
    } finally {
      setRunning(false);
      abortRef.current = null;
    }
  };

  const handleCancel = () => { abortRef.current?.abort(); };

  // ── Section 4: Env File ─────────────────────────────────────────────────
  const handleCopyEnv = async () => {
    try { await navigator.clipboard.writeText(envContents); } catch (_) {}
  };

  const handleWriteEnv = async () => {
    // The env was already written server-side via provisionEnvironment → writeEnvFile
    // This button is a confirmation/acknowledgment only
    setEnvWritten(true);
    unlockSection(null, 'envFile', 'smokeTest');
  };

  // ── Section 5: Smoke Test ───────────────────────────────────────────────
  const handleSmokeTest = async () => {
    setSmoking(true);
    setSmokeResults(null);
    try {
      const apps = provisioned ? [
        provisioned.adminApp && { name: 'Admin App', clientId: provisioned.adminApp.clientId, clientSecret: provisioned.adminApp.clientSecret },
        provisioned.userApp && { name: 'User App', clientId: provisioned.userApp.clientId, clientSecret: provisioned.userApp.clientSecret },
        provisioned.agentApp && { name: 'Agent App', clientId: provisioned.agentApp.clientId, clientSecret: provisioned.agentApp.clientSecret },
        provisioned.mcpExchangerApp && { name: 'MCP Token Exchanger', clientId: provisioned.mcpExchangerApp.clientId, clientSecret: provisioned.mcpExchangerApp.clientSecret },
      ].filter(Boolean) : [];

      const { data } = await apiClient.post('/api/setup/wizard/smoke-test', {
        envId: creds.envId,
        workerClientId: creds.workerClientId,
        workerClientSecret: creds.workerClientSecret,
        region: creds.region,
        apps,
      });
      setSmokeResults(data.results || []);
      if (data.ok) {
        setSections(s => ({ ...s, smokeTest: STATUS.COMPLETE }));
      }
    } catch (err) {
      notifyError(err.response?.data?.error || err.message || 'Smoke test failed');
    } finally {
      setSmoking(false);
    }
  };

  // ── .env generation ─────────────────────────────────────────────────────
  function generateEnvContents(result, formCreds) {
    const lines = [
      '# Generated by BX Finance Setup Wizard — ' + new Date().toISOString(),
      '',
      '# PingOne Core',
      'PINGONE_ENVIRONMENT_ID=' + formCreds.envId,
      'PINGONE_REGION=' + formCreds.region,
      '',
      '# Admin App',
      'PINGONE_ADMIN_CLIENT_ID=' + (result.adminApp?.clientId || ''),
      'PINGONE_ADMIN_CLIENT_SECRET=' + (result.adminApp?.clientSecret || '<retrieve-from-pingone>'),
      'PINGONE_ADMIN_REDIRECT_URI=' + formCreds.publicAppUrl + '/api/auth/oauth/callback',
      '',
      '# User App',
      'PINGONE_CORE_CLIENT_ID=' + (result.userApp?.clientId || ''),
      'PINGONE_CORE_CLIENT_SECRET=' + (result.userApp?.clientSecret || '<retrieve-from-pingone>'),
      'PINGONE_CORE_REDIRECT_URI=' + formCreds.publicAppUrl + '/api/auth/oauth/user/callback',
      '',
      '# Worker App',
      'PINGONE_WORKER_TOKEN_CLIENT_ID=' + formCreds.workerClientId,
      'PINGONE_WORKER_TOKEN_CLIENT_SECRET=' + formCreds.workerClientSecret,
      'PINGONE_WORKER_TOKEN_AUTH_METHOD=' + formCreds.tokenAuthMethod,
      '',
      '# Agent Worker App',
      'PINGONE_AGENT_CLIENT_ID=' + (result.agentApp?.clientId || ''),
      'PINGONE_AGENT_CLIENT_SECRET=' + (result.agentApp?.clientSecret || '<retrieve-from-pingone>'),
      '',
      '# MCP Token Exchanger',
      'PINGONE_MCP_TOKEN_EXCHANGER_CLIENT_ID=' + (result.mcpExchangerApp?.clientId || ''),
      'PINGONE_MCP_TOKEN_EXCHANGER_CLIENT_SECRET=' + (result.mcpExchangerApp?.clientSecret || '<retrieve-from-pingone>'),
      '',
      '# Public URL',
      'PINGONE_PUBLIC_APP_URL=' + formCreds.publicAppUrl,
      '',
      '# MFA Step-up',
      'STEP_UP_ACR_VALUE=' + formCreds.stepUpAcrValue,
      'PINGONE_MFA_ACR_VALUE=' + formCreds.stepUpAcrValue,
    ];
    return lines.join('\n');
  }

  // ── Render ──────────────────────────────────────────────────────────────
  const toggleSection = (name) => {
    if (sections[name] === STATUS.LOCKED) return;
    setOpenSection(prev => prev === name ? null : name);
  };

  const isOpen = (name) => openSection === name;

  const allDone = Object.values(sections).every(s => s === STATUS.COMPLETE);

  return (
    <div className="swp-root">
      <div className="swp-header">
        <h1>BX Finance Setup Wizard</h1>
        <p>Configure your PingOne environment automatically. Takes about 2 minutes.</p>
        <Link to="/setup" className="swp-back-link">← Back to Setup</Link>
      </div>

      <div className="swp-accordion">

        {/* ── Section 1: Credentials ── */}
        <div className={`swp-section swp-section--${sections.credentials}`}>
          <SectionHeader title="Worker Credentials" status={sections.credentials} index={1} onToggle={() => toggleSection('credentials')} isOpen={isOpen('credentials')} />
          {isOpen('credentials') && (
            <div className="swp-section-body">
              <p className="swp-section-desc">Enter your PingOne worker app credentials. The wizard uses these to call the Management API.</p>
              <div className="swp-form-grid">
                <div className="swp-field">
                  <label htmlFor="envId">Environment ID *</label>
                  <input id="envId" type="text" value={creds.envId} onChange={e => setCreds(p => ({...p, envId: e.target.value}))} placeholder="xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx" className={credErrors.envId ? 'swp-input--error' : ''} disabled={validating} />
                  {credErrors.envId && <span className="swp-field-error">{credErrors.envId}</span>}
                </div>
                <div className="swp-field">
                  <label htmlFor="region">Region</label>
                  <select id="region" value={creds.region} onChange={e => setCreds(p => ({...p, region: e.target.value}))} disabled={validating}>
                    {REGION_OPTIONS.map(o => <option key={o.value} value={o.value}>{o.label}</option>)}
                  </select>
                </div>
                <div className="swp-field">
                  <label htmlFor="workerClientId">Worker Client ID *</label>
                  <input id="workerClientId" type="text" value={creds.workerClientId} onChange={e => setCreds(p => ({...p, workerClientId: e.target.value}))} placeholder="worker-client-id" className={credErrors.workerClientId ? 'swp-input--error' : ''} disabled={validating} />
                  {credErrors.workerClientId && <span className="swp-field-error">{credErrors.workerClientId}</span>}
                </div>
                <div className="swp-field swp-field--password">
                  <label htmlFor="workerClientSecret">Worker Client Secret *</label>
                  <div className="swp-input-row">
                    <input id="workerClientSecret" type={showSecret ? 'text' : 'password'} value={creds.workerClientSecret} onChange={e => setCreds(p => ({...p, workerClientSecret: e.target.value}))} placeholder="worker-client-secret" className={credErrors.workerClientSecret ? 'swp-input--error' : ''} disabled={validating} autoComplete="new-password" />
                    <button type="button" className="swp-show-btn" onClick={() => setShowSecret(p => !p)}>{showSecret ? 'Hide' : 'Show'}</button>
                  </div>
                  {credErrors.workerClientSecret && <span className="swp-field-error">{credErrors.workerClientSecret}</span>}
                </div>
                <div className="swp-field">
                  <label htmlFor="tokenAuthMethod">Token Auth Method</label>
                  <select id="tokenAuthMethod" value={creds.tokenAuthMethod} onChange={e => setCreds(p => ({...p, tokenAuthMethod: e.target.value}))} disabled={validating}>
                    {AUTH_METHOD_OPTIONS.map(o => <option key={o.value} value={o.value}>{o.label}</option>)}
                  </select>
                </div>
                <div className="swp-field">
                  <label htmlFor="publicAppUrl">Public App URL *</label>
                  <input id="publicAppUrl" type="url" value={creds.publicAppUrl} onChange={e => setCreds(p => ({...p, publicAppUrl: e.target.value}))} placeholder="http://localhost:3001" className={credErrors.publicAppUrl ? 'swp-input--error' : ''} disabled={validating} />
                  {credErrors.publicAppUrl && <span className="swp-field-error">{credErrors.publicAppUrl}</span>}
                </div>
                <div className="swp-field">
                  <label htmlFor="stepUpAcrValue">Step-up ACR Value <span className="swp-optional">(optional)</span></label>
                  <input id="stepUpAcrValue" type="text" value={creds.stepUpAcrValue} onChange={e => setCreds(p => ({...p, stepUpAcrValue: e.target.value}))} placeholder="Multi_Factor" disabled={validating} />
                  <small className="swp-help">PingOne Sign-On Policy name for MFA step-up. Leave blank to use app default.</small>
                </div>
              </div>
              <div className="swp-section-actions">
                <button type="button" className="swp-btn swp-btn--primary" onClick={handleValidate} disabled={validating}>
                  {validating ? 'Validating…' : 'Validate Credentials →'}
                </button>
              </div>
            </div>
          )}
        </div>

        {/* ── Section 2: Discovery ── */}
        <div className={`swp-section swp-section--${sections.discovery}`}>
          <SectionHeader title="Discover Existing Objects" status={sections.discovery} index={2} onToggle={() => toggleSection('discovery')} isOpen={isOpen('discovery')} />
          {isOpen('discovery') && (
            <div className="swp-section-body">
              <p className="swp-section-desc">Check what PingOne objects already exist. Existing objects will be skipped during creation.</p>
              {discovery ? (
                <table className="swp-discovery-table">
                  <thead><tr><th>Object</th><th>Status</th></tr></thead>
                  <tbody>
                    {Object.entries(discovery).map(([key, val]) => (
                      <tr key={key}>
                        <td>{key}</td>
                        <td className={val ? 'swp-exists' : 'swp-missing'}>{val ? '✅ Exists' : '➕ Will create'}</td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              ) : (
                <p className="swp-muted">Click Discover to probe your PingOne environment.</p>
              )}
              <div className="swp-section-actions">
                <button type="button" className="swp-btn swp-btn--primary" onClick={handleDiscover} disabled={discovering}>
                  {discovering ? 'Discovering…' : discovery ? 'Re-discover' : 'Discover →'}
                </button>
                {discovery && (
                  <button type="button" className="swp-btn swp-btn--secondary" onClick={() => unlockSection(null, 'discovery', 'create')}>
                    Continue to Create →
                  </button>
                )}
              </div>
            </div>
          )}
        </div>

        {/* ── Section 3: Create ── */}
        <div className={`swp-section swp-section--${sections.create}`}>
          <SectionHeader title="Create PingOne Objects" status={sections.create} index={3} onToggle={() => toggleSection('create')} isOpen={isOpen('create')} />
          {isOpen('create') && (
            <div className="swp-section-body">
              <p className="swp-section-desc">Create all missing apps, resource servers, scopes, and attribute mappings.</p>
              <label className="swp-toggle">
                <input type="checkbox" checked={forceRecreate} onChange={e => setForceRecreate(e.target.checked)} disabled={running} />
                <span> Reset &amp; Recreate — delete existing objects and create fresh (destructive)</span>
              </label>
              <div className="swp-log-container">
                {logEntries.length === 0 ? (
                  <p className="swp-muted">Setup log will appear here.</p>
                ) : (
                  <div className="swp-log-entries">
                    {logEntries.map((e, i) => <LogRow key={i} entry={e} />)}
                    <div ref={logEndRef} />
                  </div>
                )}
              </div>
              <div className="swp-section-actions">
                {!running ? (
                  <button type="button" className="swp-btn swp-btn--primary" onClick={handleRun} disabled={running}>
                    Run Setup →
                  </button>
                ) : (
                  <button type="button" className="swp-btn swp-btn--danger" onClick={handleCancel}>
                    Cancel
                  </button>
                )}
              </div>
            </div>
          )}
        </div>

        {/* ── Section 4: Env File ── */}
        <div className={`swp-section swp-section--${sections.envFile}`}>
          <SectionHeader title="Environment File" status={sections.envFile} index={4} onToggle={() => toggleSection('envFile')} isOpen={isOpen('envFile')} />
          {isOpen('envFile') && (
            <div className="swp-section-body">
              <p className="swp-section-desc">
                The wizard wrote your <code>.env</code> file and updated the live config. Review the generated values below.
                {envWritten && <span className="swp-badge swp-badge--ok"> ✅ Written to disk</span>}
              </p>
              <div className="swp-env-header">
                <button type="button" className="swp-btn swp-btn--secondary swp-btn--sm" onClick={handleCopyEnv}>Copy</button>
              </div>
              <textarea className="swp-env-textarea" value={envContents} readOnly rows={20} />
              <div className="swp-section-actions">
                {!envWritten && (
                  <button type="button" className="swp-btn swp-btn--primary" onClick={handleWriteEnv}>
                    Confirm &amp; Continue →
                  </button>
                )}
                {envWritten && (
                  <button type="button" className="swp-btn swp-btn--secondary" onClick={() => unlockSection(null, 'envFile', 'smokeTest')}>
                    Continue to Smoke Test →
                  </button>
                )}
              </div>
            </div>
          )}
        </div>

        {/* ── Section 5: Smoke Test ── */}
        <div className={`swp-section swp-section--${sections.smokeTest}`}>
          <SectionHeader title="Smoke Test" status={sections.smokeTest} index={5} onToggle={() => toggleSection('smokeTest')} isOpen={isOpen('smokeTest')} />
          {isOpen('smokeTest') && (
            <div className="swp-section-body">
              <p className="swp-section-desc">Verify each worker app can obtain a token from PingOne.</p>
              {smokeResults && (
                <div className="swp-smoke-results">
                  {smokeResults.map((r, i) => (
                    <div key={i} className={`swp-smoke-row swp-smoke-row--${r.success ? 'pass' : 'fail'}`}>
                      <span>{r.success ? '✅' : '❌'}</span>
                      <strong>{r.app}</strong>
                      <span>{r.success ? 'Token obtained' : r.error}</span>
                    </div>
                  ))}
                </div>
              )}
              <div className="swp-section-actions">
                <button type="button" className="swp-btn swp-btn--primary" onClick={handleSmokeTest} disabled={smoking}>
                  {smoking ? 'Testing…' : smokeResults ? 'Re-run Tests' : 'Run Tests →'}
                </button>
              </div>
            </div>
          )}
        </div>

      </div>

      {/* ── Completion Banner ── */}
      {allDone && smokeResults?.every(r => r.success) && (
        <div className="swp-complete-banner">
          <h2>🎉 Setup Complete!</h2>
          <p>Your BX Finance demo is ready to run.</p>

            <a href="/" className="swp-btn swp-btn--primary">Open App →</a>
            <a href="/api/auth/oauth/login" className="swp-btn swp-btn--secondary">Admin Login →</a>
          </div>
        </div>
      )}
    </div>
  );
}
```
