## Task 3: Add standalone POST /api/setup/wizard/write-env endpoint

<read_first>
- banking_api_server/routes/setupWizard.js (publicRouter — add to it)
- banking_api_server/services/configStore.js (setConfig signature)
</read_first>

<action>
In `setupWizard.js` `publicRouter`, add a new endpoint:

```javascript
// POST /api/setup/wizard/write-env — merge key-value pairs into .env, also write configStore
publicRouter.post('/write-env', wizardRateLimit, async (req, res) => {
  const { envVars } = req.body; // { key: value, ... }
  if (!envVars || typeof envVars !== 'object') {
    return res.status(400).json({ error: 'missing_env_vars', message: 'envVars object required' });
  }

  const isVercel = !!process.env.VERCEL;
  const results = { written: [], configStored: [], skipped: [] };

  // configStore integration
  let configStore;
  try { configStore = require('../services/configStore'); } catch (_) {}
  if (configStore) {
    for (const [key, value] of Object.entries(envVars)) {
      if (value) {
        try { configStore.setConfig(key.toLowerCase(), value); results.configStored.push(key); }
        catch (_) {}
      }
    }
  }

  // .env file merge (local only)
  if (!isVercel) {
    const fs = require('fs').promises;
    const path = require('path');
    const envPath = path.resolve(__dirname, '../../../.env');
    try {
      let existing = {};
      try {
        const contents = await fs.readFile(envPath, 'utf8');
        for (const line of contents.split('\n')) {
          const m = line.match(/^([A-Z_][A-Z0-9_]*)=(.*)$/);
          if (m) existing[m[1]] = m[2];
        }
      } catch (_) { /* file doesn't exist yet */ }

      for (const [key, value] of Object.entries(envVars)) {
        existing[key] = value;
        results.written.push(key);
      }

      const lines = Object.entries(existing).map(([k, v]) => k + '=' + v);
      await fs.writeFile(envPath, lines.join('\n') + '\n', 'utf8');
    } catch (writeErr) {
      return res.status(500).json({ error: 'write_failed', message: writeErr.message });
    }
  } else {
    results.skipped = Object.keys(envVars);
  }

  res.json({ ok: true, isVercel, results });
});
```
</action>

<acceptance_criteria>
- `POST /api/setup/wizard/write-env` accepts `{ envVars: { KEY: value } }`
- Returns `{ ok: true, isVercel, results: { written, configStored, skipped } }`
- On Vercel (`isVercel === true`), skips file write but still calls configStore
- On local, reads existing `.env`, merges (preserves existing keys not in `envVars`), writes back
- Returns 400 if `envVars` missing, 500 if file write fails
- No `requireAdmin` guard
</acceptance_criteria>
