**2. Add scope injection helper function (near top of file after other helpers):**

```javascript
/**
 * Inject banking scopes when conditionally enabled and user token lacks them.
 * Pattern: mirrors ff_inject_may_act and ff_inject_audience injection pattern.
 * 
 * @param {object} claims — decoded JWT claims from user token
 * @param {Array} tokenEvents — accumulator for UI display
 * @returns {object} modified claims (or original if injection not needed)
 */
function conditionalInjectBankingScopes(claims, tokenEvents) {
  if (!claims || claims.scope == null) {
    return claims; // No scope claim to check
  }

  const scopes = String(claims.scope).trim().split(/\s+/).filter(Boolean);
  const hasBankingScopes = scopes.some(s => s.startsWith('banking:') || s === 'banking:read' || s === 'banking:write');
  
  if (hasBankingScopes) {
    return claims; // Already has banking scopes, no injection needed
  }

  // Check if injection is enabled
  const shouldInjectScopes = 
    configStore.getEffective('ff_inject_scopes') === true ||
    configStore.getEffective('ff_inject_scopes') === 'true';

  if (!shouldInjectScopes) {
    return claims; // Injection not enabled
  }

  // Inject banking:read and banking:write
  const injectScopes = ['banking:read', 'banking:write'];
  claims.scope = (claims.scope || '') + ' ' + injectScopes.join(' ');
  
  // Track in tokenEvents for UI transparency
  tokenEvents.push({
    timestamp: new Date().toISOString(),
    phase: 'Token Claim Injection',
    label: '[BFF-INJECTED scope]',
    detail: `Scopes synthetically injected: ${injectScopes.join(', ')}. Enable ff_inject_scopes is ON. ` +
            `User access token had no banking scopes so the BFF has injected demo scopes. ` +
            `(Normally scopes come from PingOne; injection for demo/dev only.)`,
    injected: true,
  });

  errorLog(
    `[SCOPE_INJECTION] Banking scopes injected: ${injectScopes.join(', ')}. ` +
    `User token had no banking scopes. ff_inject_scopes = true (demo mode).`
  );

  return claims;
}
```
