**Add GET endpoint to serve SCOPE_VOCABULARY.md content.**

**1. In adminConfig.js, add route after existing routes:**

```javascript
/**
 * GET /api/admin/scope-vocabulary
 * Serves the canonical scope vocabulary registry for UI display.
 * Returns markdown content as JSON with optional HTML rendering.
 */
router.get('/scope-vocabulary', authenticateToken, requireScopes(['banking:admin']), async (req, res, next) => {
  try {
    const path = require('path');
    const fs = require('fs').promises;
    
    // Read SCOPE_VOCABULARY.md from banking_api_server/
    const vocabPath = path.join(__dirname, '..', 'SCOPE_VOCABULARY.md');
    const markdownContent = await fs.readFile(vocabPath, 'utf-8');
    
    // Optional: Convert markdown to HTML (if marked library available)
    let htmlContent = null;
    try {
      const marked = require('marked');
      htmlContent = marked(markdownContent);
    } catch (err) {
      // marked not available, return markdown only
    }
    
    res.json({
      success: true,
      markdown: markdownContent,
      html: htmlContent,
      timestamp: new Date().toISOString()
    });
    
  } catch (error) {
    errorLog('[ScopeVocabularyEndpoint]', error.message);
    res.status(500).json({
      success: false,
      error: 'Failed to retrieve scope vocabulary',
      message: error.message
    });
  }
});
```

**2. Error handling:**
- If file not found: return 404 with message
- If read fails: return 500 with message
- All errors logged to console

**3. Security:**
- Requires authentication (authenticateToken middleware)
- Requires admin scope (requireScopes(['banking:admin']))
- Only admins can view scope vocabulary

**4. Optional: Cache content:**

```javascript
// At module level:
let cachedVocabulary = null;
let vocabCacheTime = null;
const VOCAB_CACHE_TTL = 5 * 60 * 1000; // 5 minutes

// In route, before reading file:
if (cachedVocabulary && (Date.now() - vocabCacheTime) < VOCAB_CACHE_TTL) {
  return res.json(cachedVocabulary);
}

// After reading, before return:
cachedVocabulary = { success: true, markdown, html, timestamp };
vocabCacheTime = Date.now();
```

**Option A: Extract modal to shared component (recommended for reuse)**

Create new component: `banking_api_ui/src/components/ScopeVocabularyModal.jsx`

```jsx
import React, { useState, useEffect } from 'react';

export const ScopeVocabularyModal = ({ isOpen, onClose }) => {
  const [content, setContent] = useState('');
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    if (isOpen) {
      loadVocabulary();
    }
  }, [isOpen]);

  const loadVocabulary = async () => {
    setLoading(true);
    try {
      const res = await fetch('/api/admin/scope-vocabulary');
      if (res.ok) {
        const data = await res.json();
        setContent(data.markdown || data.html || '');
      }
    } catch (error) {
      console.error('Error loading scope vocabulary:', error);
    } finally {
      setLoading(false);
    }
  };

  if (!isOpen) return null;

  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className="modal-content" onClick={e => e.stopPropagation()}>
        <div className="modal-header">
          <h2>Banking Scopes — Canonical Reference</h2>
          <button className="modal-close" onClick={onClose}>✕</button>
        </div>
        <div className="modal-body markdown-content">
          {loading ? <p>Loading...</p> : <div dangerouslySetInnerHTML={{ __html: content }} />}
        </div>
      </div>
    </div>
  );
};
```
