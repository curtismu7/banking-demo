## Implementation Patterns

### Agent Authentication Pattern
```javascript
// Agent registration and authentication
const agentConfig = {
  clientId: 'ai-agent-banking-assistant',
  clientSecret: process.env.AI_AGENT_SECRET,
  grant_type: 'client_credentials',
  scope: 'banking:read banking:write ai:act'
};

// Token exchange for human delegation
const delegationConfig = {
  subject_token: userAccessToken,
  subject_token_type: 'urn:ietf:params:oauth:token-type:access_token',
  actor_token: agentToken,
  actor_token_type: 'urn:ietf:params:oauth:token-type:access_token',
  requested_token_type: 'urn:ietf:params:oauth:token-type:access_token',
  audience: 'banking-api',
  scope: 'banking:read transactions:read'
};
```

### Human-in-the-Loop Pattern
```javascript
// Step-up authentication for sensitive AI actions
const sensitiveAction = async (action, context) => {
  // Check if step-up required
  if (requiresStepUp(action, context)) {
    await triggerStepUpMFA(context.userId);
  }
  
  // Log human approval
  await auditService.log({
    action: 'ai_sensitive_operation',
    userId: context.userId,
    agentId: context.agentId,
    operation: action,
    approved: true,
    timestamp: new Date()
  });
  
  // Execute with human authority
  return executeWithDelegation(action, context);
};
```

### Multi-Agent Coordination Pattern
```javascript
// Coordinated agent actions with audit correlation
const coordinatedAction = async (agents, operation) => {
  const correlationId = generateCorrelationId();
  const results = [];
  
  for (const agent of agents) {
    const result = await agent.execute(operation, {
      correlationId,
      delegationChain: buildDelegationChain(agent),
      auditContext: {
        operation,
        agents: agents.map(a => a.id),
        coordinator: 'lead-agent'
      }
    });
    
    results.push(result);
  }
  
  return {
    correlationId,
    results,
    auditTrail: await auditService.getCorrelation(correlationId)
  };
};
```
