    ```js
    // banking_api_server/routes/langchainConfig.js
    'use strict';
    const express = require('express');
    const router = express.Router();

    const PROVIDER_MODELS = {
      openai:    ['gpt-4o', 'gpt-4o-mini', 'gpt-3.5-turbo'],
      groq:      ['llama-3.1-8b-instant', 'llama-3.3-70b-versatile', 'mixtral-8x7b-32768'],
      anthropic: ['claude-3-5-sonnet-20241022', 'claude-3-haiku-20240307'],
      google:    ['gemini-1.5-flash', 'gemini-1.5-pro'],
      ollama:    ['llama3.2', 'mistral', 'phi3'],
    };

    const DEFAULT_MODELS = {
      openai: 'gpt-4o-mini', groq: 'llama-3.1-8b-instant',
      anthropic: 'claude-3-5-sonnet-20241022', google: 'gemini-1.5-flash', ollama: 'llama3.2',
    };

    function getLangchainConfig(session) {
      return session.langchain_config || {
        provider: process.env.LANGCHAIN_PROVIDER || 'groq',
        model: DEFAULT_MODELS[process.env.LANGCHAIN_PROVIDER || 'groq'],
        keys: {},
      };
    }

    // GET /api/langchain/config/status — current config, keys as booleans (never return values)
    router.get('/config/status', (req, res) => {
      const cfg = getLangchainConfig(req.session);
      const keySet = {
        openai:          !!cfg.keys?.openai,
        groq:            !!(cfg.keys?.groq || process.env.GROQ_API_KEY),
        anthropic:       !!cfg.keys?.anthropic,
        google:          !!cfg.keys?.google,
        ollama_base_url: !!cfg.keys?.ollama_base_url,
      };
      return res.json({
        provider: cfg.provider,
        model: cfg.model || DEFAULT_MODELS[cfg.provider] || '',
        key_set: keySet,
        provider_models: PROVIDER_MODELS,
      });
    });

    // POST /api/langchain/config — save provider/model/key to session only
    router.post('/config', (req, res) => {
      const { provider, model, key, key_type } = req.body || {};
      if (!provider || !PROVIDER_MODELS[provider]) {
        return res.status(400).json({ error: 'invalid_provider', message: `provider must be one of: ${Object.keys(PROVIDER_MODELS).join(', ')}` });
      }
      const cfg = getLangchainConfig(req.session);
      cfg.provider = provider;
      cfg.model = model || DEFAULT_MODELS[provider] || '';
      if (key && key_type) {
        cfg.keys = cfg.keys || {};
        cfg.keys[key_type] = key; // stored in session, never logged or returned
      }
      req.session.langchain_config = cfg;
      return res.json({
        ok: true,
        provider: cfg.provider,
        model: cfg.model,
        key_set: !!key,
      });
    });

    // DELETE /api/langchain/config/key/:provider — clear a specific key from session
    router.delete('/config/key/:keyType', (req, res) => {
      const { keyType } = req.params;
      const cfg = getLangchainConfig(req.session);
      if (cfg.keys) delete cfg.keys[keyType];
      req.session.langchain_config = cfg;
      return res.json({ ok: true, cleared: keyType });
    });

    module.exports = router;
    ```

    **Add a new `LangChainAgentConfig` sub-component** near the top of the file (before the main `Config` function), following the same inline component pattern used by other sections:

    ```jsx
    function LangChainAgentConfig() {
      const PROVIDERS = [
        { id: 'groq',      label: 'Groq',      keyLabel: 'GROQ_API_KEY',      keyType: 'groq',      placeholder: 'gsk_...' },
        { id: 'openai',    label: 'OpenAI',    keyLabel: 'OPENAI_API_KEY',    keyType: 'openai',    placeholder: 'sk-...' },
        { id: 'anthropic', label: 'Anthropic', keyLabel: 'ANTHROPIC_API_KEY', keyType: 'anthropic', placeholder: 'sk-ant-...' },
        { id: 'google',    label: 'Gemini',    keyLabel: 'GOOGLE_API_KEY',    keyType: 'google',    placeholder: 'AIzaSy...' },
        { id: 'ollama',    label: 'Ollama',    keyLabel: 'Base URL',          keyType: 'ollama_base_url', placeholder: 'http://localhost:11434' },
      ];
      const [status, setStatus] = React.useState(null);
      const [provider, setProvider] = React.useState('groq');
      const [model, setModel] = React.useState('');
      const [keyValues, setKeyValues] = React.useState({});
      const [saving, setSaving] = React.useState(false);
      const [msg, setMsg] = React.useState('');

      React.useEffect(() => {
        fetch('/api/langchain/config/status', { credentials: 'include' })
          .then(r => r.json())
          .then(d => {
            setStatus(d);
            setProvider(d.provider || 'groq');
            setModel(d.model || '');
          })
          .catch(() => {});
      }, []);

      const providerModels = status?.provider_models || {};

      async function handleSave(keyType) {
        setSaving(true); setMsg('');
        try {
          const body = { provider, model, key: keyValues[keyType] || '', key_type: keyType };
          const res = await fetch('/api/langchain/config', {
            method: 'POST', credentials: 'include',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(body),
          });
          if (res.ok) {
            setKeyValues(v => ({ ...v, [keyType]: '' })); // clear field after save
            const d = await res.json();
            setStatus(prev => ({ ...prev, key_set: { ...(prev?.key_set || {}), [keyType]: d.key_set } }));
            setMsg('✅ Saved to session');
          }
        } finally { setSaving(false); }
      }

      async function handleClear(keyType) {
        await fetch(`/api/langchain/config/key/${keyType}`, { method: 'DELETE', credentials: 'include' });
        setStatus(prev => ({ ...prev, key_set: { ...(prev?.key_set || {}), [keyType]: false } }));
      }

      return (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
          <p style={{ fontSize: '0.875rem', color: '#6b7280', margin: 0 }}>
            API keys are stored <strong>server-side only</strong> (session) — consistent with BFF token security. Keys are cleared on session expiry.
          </p>

          <div style={{ display: 'flex', gap: 12, alignItems: 'center' }}>
            <label style={{ fontWeight: 600, fontSize: '0.875rem' }}>Default provider:</label>
            <select value={provider} onChange={e => { setProvider(e.target.value); setModel(''); }}
              style={{ padding: '4px 8px', borderRadius: 4, border: '1px solid #d1d5db' }}>
              {PROVIDERS.map(p => <option key={p.id} value={p.id}>{p.label}</option>)}
            </select>
            <select value={model} onChange={e => setModel(e.target.value)}
              style={{ padding: '4px 8px', borderRadius: 4, border: '1px solid #d1d5db', minWidth: 200 }}>
              <option value="">— select model —</option>
              {(providerModels[provider] || []).map(m => <option key={m} value={m}>{m}</option>)}
            </select>
          </div>

          {PROVIDERS.map(p => (
            <div key={p.id} style={{ display: 'flex', flexDirection: 'column', gap: 4 }}>
              <div style={{ display: 'flex', gap: 8, alignItems: 'center' }}>
                <span style={{ fontWeight: 600, fontSize: '0.8rem', width: 80 }}>{p.label}</span>
                {status?.key_set?.[p.keyType]
                  ? <span style={{ color: '#059669', fontSize: '0.8rem' }}>🔒 key set (session only)</span>
                  : null}
                {status?.key_set?.[p.keyType]
                  ? <button onClick={() => handleClear(p.keyType)} style={{ fontSize: '0.75rem', color: '#6b7280', cursor: 'pointer', background: 'none', border: 'none', textDecoration: 'underline' }}>Clear</button>
                  : (
                    <>
                      <input type="password" placeholder={p.placeholder}
                        value={keyValues[p.keyType] || ''}
                        onChange={e => setKeyValues(v => ({ ...v, [p.keyType]: e.target.value }))}
                        style={{ flex: 1, padding: '4px 8px', borderRadius: 4, border: '1px solid #d1d5db', fontFamily: 'monospace', fontSize: '0.8rem' }} />
                      <button onClick={() => handleSave(p.keyType)} disabled={saving || !keyValues[p.keyType]}
                        style={{ padding: '4px 12px', borderRadius: 4, background: '#4f7df3', color: '#fff', border: 'none', cursor: 'pointer', fontSize: '0.8rem' }}>
                        Save
                      </button>
                    </>
                  )}
              </div>
            </div>
          ))}
          {msg && <p style={{ color: '#059669', fontSize: '0.8rem', margin: 0 }}>{msg}</p>}
        </div>
      );
    }
    ```
