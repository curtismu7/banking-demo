    ```jsx
    const PROVIDER_MODELS = {
      openai:    ['gpt-4o', 'gpt-4o-mini', 'gpt-3.5-turbo'],
      groq:      ['llama-3.1-8b-instant', 'llama-3.3-70b-versatile', 'mixtral-8x7b-32768'],
      anthropic: ['claude-3-5-sonnet-20241022', 'claude-3-haiku-20240307'],
      google:    ['gemini-1.5-flash', 'gemini-1.5-pro'],
      ollama:    ['llama3.2', 'mistral', 'phi3'],
    };
    const PROVIDER_LABELS = { openai: 'OpenAI', groq: 'Groq', anthropic: 'Anthropic', google: 'Gemini', ollama: 'Ollama' };
    ```

    Inside `ChatWidget` function, add state:
    ```jsx
    const [activeProvider, setActiveProvider] = useState('groq');
    const [activeModel, setActiveModel] = useState('llama-3.1-8b-instant');
    const [showSettings, setShowSettings] = useState(false);
    const [settingsProvider, setSettingsProvider] = useState('groq');
    const [settingsModel, setSettingsModel] = useState('');
    const [settingsKey, setSettingsKey] = useState('');
    const [keySet, setKeySet] = useState({});
    const [settingsSaving, setSettingsSaving] = useState(false);
    const [settingsMsg, setSettingsMsg] = useState('');
    ```

    Add `useEffect` to fetch active config on mount:
    ```jsx
    useEffect(() => {
      fetch('/api/langchain/config/status', { credentials: 'include' })
        .then(r => r.ok ? r.json() : null)
        .then(d => {
          if (d) {
            setActiveProvider(d.provider || 'groq');
            setActiveModel(d.model || 'llama-3.1-8b-instant');
            setSettingsProvider(d.provider || 'groq');
            setSettingsModel(d.model || '');
            setKeySet(d.key_set || {});
          }
        })
        .catch(() => {});
    }, []);
    ```

    Add `handleSettingsSave` function:
    ```jsx
    const handleSettingsSave = async () => {
      setSettingsSaving(true); setSettingsMsg('');
      try {
        const keyType = settingsProvider === 'ollama' ? 'ollama_base_url' : settingsProvider;
        const res = await fetch('/api/langchain/config', {
          method: 'POST', credentials: 'include',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ provider: settingsProvider, model: settingsModel, key: settingsKey, key_type: keyType }),
        });
        if (res.ok) {
          setActiveProvider(settingsProvider);
          setActiveModel(settingsModel || PROVIDER_MODELS[settingsProvider]?.[0] || '');
          setSettingsKey('');
          setKeySet(prev => ({ ...prev, [keyType]: !!settingsKey }));
          setSettingsMsg('✅ Saved');
          setTimeout(() => { setSettingsMsg(''); setShowSettings(false); }, 1200);
        }
      } finally { setSettingsSaving(false); }
    };
    ```

    In the JSX `chat-header` div, ADD before the minimize button:
    ```jsx
    <button
      className="lc-badge"
      onClick={() => setShowSettings(s => !s)}
      title="Switch LLM provider/model"
    >
      ⚡ {PROVIDER_LABELS[activeProvider] || activeProvider} · {activeModel}
    </button>
    ```

    Below the header div (but inside `.chat-widget`), ADD:
    ```jsx
    {showSettings && (
      <div className="lc-settings-panel">
        <div className="lc-settings-panel__title">LLM Provider Settings</div>
        <div className="lc-settings-panel__row">
          <label>Provider</label>
          <select value={settingsProvider} onChange={e => { setSettingsProvider(e.target.value); setSettingsModel(''); }}>
            {Object.keys(PROVIDER_MODELS).map(p => (
              <option key={p} value={p}>{PROVIDER_LABELS[p]}</option>
            ))}
          </select>
        </div>
        <div className="lc-settings-panel__row">
          <label>Model</label>
          <select value={settingsModel} onChange={e => setSettingsModel(e.target.value)}>
            <option value="">— default —</option>
            {(PROVIDER_MODELS[settingsProvider] || []).map(m => (
              <option key={m} value={m}>{m}</option>
            ))}
          </select>
        </div>
        <div className="lc-settings-panel__row">
          <label>{settingsProvider === 'ollama' ? 'Base URL' : 'API Key'}</label>
          {keySet[settingsProvider === 'ollama' ? 'ollama_base_url' : settingsProvider]
            ? <span className="lc-key-set">🔒 key set (session only)</span>
            : <input
                type="password"
                placeholder={settingsProvider === 'ollama' ? 'http://localhost:11434' : 'Paste key…'}
                value={settingsKey}
                onChange={e => setSettingsKey(e.target.value)}
              />}
        </div>
        <div className="lc-settings-panel__row lc-settings-panel__row--actions">
          <button className="lc-btn-save" onClick={handleSettingsSave} disabled={settingsSaving}>
            {settingsSaving ? 'Saving…' : 'Apply'}
          </button>
          <a href="/langchain" className="lc-learn-more" target="_blank" rel="noreferrer">
            Learn more about LangChain →
          </a>
        </div>
        {settingsMsg && <p className="lc-settings-msg">{settingsMsg}</p>}
      </div>
    )}
    ```
