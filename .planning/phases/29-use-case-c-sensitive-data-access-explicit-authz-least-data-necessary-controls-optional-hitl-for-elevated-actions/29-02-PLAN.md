**Implementation detail for `checkSensitiveAccess`:**

```js
const SESSION_KEY = 'sensitiveReadConsent';
const CONSENT_TTL_MS = 60 * 1000;

async function checkSensitiveAccess(req) {
  const userId = req.user?.sub;
  if (!userId) return { allowed: false, denied: true, reason: 'no_user' };

  // 1. Check session consent
  const consent = req.session?.[SESSION_KEY];
  const consentValid = consent &&
    consent.userId === userId &&
    Date.now() < consent.expiresAt;

  // 2. Scope check (only if no valid session consent)
  if (!consentValid) {
    const tokenScopes = (req.user?.scope || req.user?.scp || '').split(' ');
    const hasScope = tokenScopes.includes('banking:sensitive:read') ||
      tokenScopes.includes('banking:read');
    if (!hasScope) {
      return { allowed: false, consent_required: true, reason: 'sensitive_data_access' };
    }
  }

  // 3. PAZ evaluation (FAIL_OPEN=false — deny if unreachable or not configured)
  try {
    const result = await evaluateSensitiveDataPaz(req);
    if (result.denied) return { allowed: false, denied: true, reason: result.reason };
  } catch (err) {
    return { allowed: false, denied: true, reason: 'paz_error' };
  }

  return { allowed: true };
}
```

**PAZ evaluation helper `evaluateSensitiveDataPaz(req)`:**
Use the same `pingOneAuthorizeService` / `simulatedAuthorizeService` pattern as
`mcpToolAuthorizationService.js`. Check `configStore.get('ff_authorize_mcp_first_tool')`
to decide if PAZ gate is enabled at all. If the PAZ flag is off, permit (the feature flag
controls the whole MCP PAZ gate, not just first-tool). If flag is on:
- Simulated mode (`simulatedAuthorizeService.isSimulatedModeEnabled(configStore)`) →

- Live mode and PAZ ready → call `pingOneAuthorizeService` with `action: 'sensitive_data_access'`
- Live mode and PAZ NOT ready (endpoint not configured) → FAIL_OPEN=false → deny with
  `reason: 'paz_not_configured'`

```js
'use strict';
const router = require('express').Router();
const dataStore = require('../data/store');
const sensitiveDataService = require('../services/sensitiveDataService');

/**
 * POST /api/accounts/sensitive-consent
 * Called by the UI when user clicks "Reveal". Grants the 60s session consent token.
 * Requires authentication (authenticateToken middleware applied at server.js registration).
 */
router.post('/sensitive-consent', async (req, res) => {
  try {
    const result = await sensitiveDataService.grantSensitiveConsent(req);
    return res.json(result);
  } catch (err) {
    return res.status(500).json({ error: 'Failed to grant consent', message: err.message });
  }
});

/**
 * GET /api/accounts/sensitive-details
 * Returns sensitive fields (accountNumberFull, routingNumber) for the authenticated user's accounts.
 * Gate: scope(banking:sensitive:read) + PAZ + session consent token.
 * On gate failure returns { ok: false, consent_required: true, reason: 'sensitive_data_access' }.
 */
router.get('/sensitive-details', async (req, res) => {
  try {
    const check = await sensitiveDataService.checkSensitiveAccess(req);

    if (!check.allowed) {
      if (check.consent_required) {
        return res.status(403).json({
          ok: false,
          consent_required: true,
          reason: check.reason || 'sensitive_data_access',
        });
      }
      return res.status(403).json({
        ok: false,
        denied: true,
        reason: check.reason || 'paz_denied',
      });
    }

    const userId = req.user.sub;
    const accounts = dataStore.getAccountsByUserId(userId);

    const sensitiveAccounts = accounts.map(account => ({
      id: account.id,
      accountType: account.accountType,
      name: account.name,
      accountNumber: account.accountNumber,       // masked ****XXXX
      accountNumberFull: account.accountNumberFull,  // exposed here only
      routingNumber: account.routingNumber,           // exposed here only
      swiftCode: account.swiftCode,
      iban: account.iban,
    }));

    return res.json({ ok: true, accounts: sensitiveAccounts });
  } catch (err) {
    return res.status(500).json({ error: 'Internal error', message: err.message });
  }
});

module.exports = router;
```
