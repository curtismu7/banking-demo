    **Approach:** AuditLogger lives in banking_mcp_server (a separate Node package).
    The BFF (banking_api_server) cannot directly `require()` it without cross-package coupling.
    Instead, add an internal HTTP endpoint to the MCP server and have the BFF proxy to it.

    **Step 1: Add internal /audit endpoint to MCP server HTTP transport**
    In banking_mcp_server/src/server/HttpMCPTransport.ts (already modified in Plan 32-01),
    add a handler for `GET /audit` (internal — protected by checking the Origin is localhost
    or the request comes from the BFF IP, or simply leave it unprotected since the MCP server
    port is not publicly exposed):

    ```typescript
    if (pathname === '/audit') {
      if (req.method !== 'GET') {
        res.writeHead(405, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ error: 'method_not_allowed' }));
        return;
      }
      const url = new URL(req.url!, 'http://localhost');
      const filters = {
        eventType: url.searchParams.get('eventType') || undefined,
        outcome: url.searchParams.get('outcome') || undefined,
        limit: url.searchParams.has('limit') ? parseInt(url.searchParams.get('limit')!, 10) : 200,
        since: url.searchParams.has('since') ? new Date(url.searchParams.get('since')!) : undefined,
      };
      const summary = url.searchParams.get('summary') === '1';
      const logger = AuditLogger.getInstance();
      if (summary) {
        const summaryData = await logger.generateAuditSummary();
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify(summaryData));
      } else {
        const events = await logger.queryAuditLogs(filters);
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify(events));
      }
      return;
    }
    ```
    Add `import { AuditLogger } from '../utils/AuditLogger';` at the top of HttpMCPTransport.ts.

    **Step 2: Create banking_api_server/routes/mcpAudit.js**
    New file at banking_api_server/routes/mcpAudit.js:

    ```javascript
    const express = require('express');
    const router = express.Router();
    const { getMcpServerUrl } = require('../utils/mcpServerConfig');  // adjust import path

    // Helper — convert ws/wss URL to http/https
    function mcpHttpBase() {
      return getMcpServerUrl().replace(/^ws:\/\//, 'http://').replace(/^wss:\/\//, 'https://');
    }

    // GET /api/mcp/audit — proxy to MCP server internal /audit endpoint
    router.get('/', async (req, res) => {
      try {
        const base = mcpHttpBase();
        const params = new URLSearchParams();
        if (req.query.eventType) params.set('eventType', req.query.eventType);
        if (req.query.outcome) params.set('outcome', req.query.outcome);
        if (req.query.limit) params.set('limit', req.query.limit);
        if (req.query.since) params.set('since', req.query.since);
        if (req.query.summary) params.set('summary', req.query.summary);

        const url = `${base}/audit${params.size ? '?' + params.toString() : ''}`;
        const upstream = await fetch(url, {
          headers: { 'Accept': 'application/json' }
        });
        if (!upstream.ok) {
          return res.status(502).json({
            error: 'mcp_audit_unavailable',
            message: `MCP server returned ${upstream.status}`
          });
        }
        const data = await upstream.json();
        res.json(data);
      } catch (err) {
        console.error('[mcpAudit] fetch error:', err.message);
        res.status(502).json({ error: 'mcp_audit_unavailable', message: err.message });
      }
    });

    module.exports = router;
    ```

    **Note on getMcpServerUrl:** Look for how it's used elsewhere in banking_api_server

    If it's a local function rather than an import, inline it:
    ```javascript

      return process.env.MCP_SERVER_URL || 'ws://localhost:8080';

    ```
