### Resource Selection UI
```javascript
// Resource selector component
const ResourceSelector = ({ selectedResources, onSelectionChange, maxResources = 3 }) => {
  const [availableResources, setAvailableResources] = useState([]);
  const [errors, setErrors] = useState([]);

  const handleResourceToggle = (resourceUri) => {
    const isSelected = selectedResources.includes(resourceUri);
    
    if (isSelected) {
      // Remove resource
      onSelectionChange(selectedResources.filter(r => r !== resourceUri));
    } else if (selectedResources.length < maxResources) {
      // Add resource
      onSelectionChange([...selectedResources, resourceUri]);
    } else {
      setErrors(['Maximum resources reached']);
    }
  };

  return (
    <div className="resource-selector">
      <h3>Select Resources to Access</h3>
      <div className="resource-grid">
        {Object.entries(resourceDefinitions).map(([uri, definition]) => (
          <ResourceCard
            key={uri}
            uri={uri}
            definition={definition}
            selected={selectedResources.includes(uri)}
            onToggle={handleResourceToggle}
          />
        ))}
      </div>
      {errors.length > 0 && (
        <div className="resource-errors">
          {errors.map(error => (
            <div key={error} className="error">{error}</div>
          ))}
        </div>
      )}
    </div>
  );
};
```

## Security Implementation

### Resource Binding
```javascript
// Cryptographic resource binding
const createResourceBinding = (token, resources) => {
  const binding = {
    resources: resources.sort(),
    client_id: token.client_id,
    user_id: token.sub,
    timestamp: Math.floor(Date.now() / 1000)
  };
  
  // Create hash for binding
  const bindingHash = crypto
    .createHash('sha256')
    .update(JSON.stringify(binding))
    .digest('hex');
  
  return {
    ...token,
    resource_binding: bindingHash
  };
};
```

### Resource Validation Middleware
```javascript
// Resource validation middleware
const resourceValidationMiddleware = (req, res, next) => {
  const token = req.user; // From authentication middleware
  const requestResource = getResourceFromRequest(req);
  
  if (!validateResourceBinding(token, requestResource)) {
    return res.status(403).json({
      error: 'invalid_resource_binding',
      message: 'Token not bound to requested resource'
    });
  }
  
  // Add resource context to request
  req.resource = requestResource;
  next();
};
```

### Cross-Resource Prevention
```javascript
// Prevent cross-resource token usage
const validateCrossResourceUsage = (token, targetResource) => {
  const allowedResources = token.resource || token.aud || [];
  
  // Strict validation - token must be bound to exact resource
  if (!allowedResources.includes(targetResource)) {
    throw new Error('Token not authorized for this resource');
  }
  
  // Additional validation for resource patterns
  const resourcePattern = /^https:\/\/([^.]+)\.example\.com\/(.*)$/;
  const tokenMatch = allowedResources[0]?.match(resourcePattern);
  const targetMatch = targetResource?.match(resourcePattern);
  
  // Prevent subdomain/resource crossing
  if (tokenMatch && targetMatch && tokenMatch[1] !== targetMatch[1]) {
    throw new Error('Cross-resource access not allowed');
  }
};
```

### Resource-Scope Validation
```javascript
// Validate that scopes are compatible with resources
const validateScopeResourceCompatibility = (scopes, resources) => {
  const resourceScopeMap = {
    'https://banking-api.example.com/': ['banking:', 'transactions:', 'accounts:'],
    'https://mcp-server.example.com/': ['ai:', 'agent:', 'mcp:'],
    'https://admin-api.example.com/': ['admin:', 'users:', 'config:'],
    'https://config-api.example.com/': ['config:', 'settings:']
  };
  
  const validScopes = [];
  
  for (const scope of scopes) {
    const isValidForResource = resources.some(resource => {
      const allowedPrefixes = resourceScopeMap[resource];
      return allowedPrefixes?.some(prefix => scope.startsWith(prefix));
    });
    
    if (isValidForResource) {
      validScopes.push(scope);
    }
  }
  
  return validScopes;
};
```
