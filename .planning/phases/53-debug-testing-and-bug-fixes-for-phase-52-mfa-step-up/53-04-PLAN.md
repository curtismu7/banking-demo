    ```js
    const handleEnrollFido2 = useCallback(async () => {
      if (!window.PublicKeyCredential) {
        setEnrollError('Your browser does not support passkeys. Please use Email OTP instead.');
        return;
      }
      setEnrolling(true);
      setEnrollError('');
      try {
        // Step 1: initiate registration
        const initRes = await apiClient.post('/api/auth/mfa/enroll/fido2-init');
        const { deviceId, publicKeyCredentialCreationOptions } = initRes.data;

        // Step 2: browser WebAuthn ceremony
        const credential = await navigator.credentials.create({
          publicKey: publicKeyCredentialCreationOptions,
        });

        // Step 3: encode and complete
        const attestation = {
          id: credential.id,
          rawId: btoa(String.fromCharCode(...new Uint8Array(credential.rawId))),
          type: credential.type,
          response: {
            attestationObject: btoa(String.fromCharCode(...new Uint8Array(credential.response.attestationObject))),
            clientDataJSON: btoa(String.fromCharCode(...new Uint8Array(credential.response.clientDataJSON))),
          },
        };
        await apiClient.post('/api/auth/mfa/enroll/fido2-complete', { deviceId, attestation });
        setEnrollModalOpen(false);
        if (handleInitiateOtpRef.current) handleInitiateOtpRef.current();
      } catch (err) {
        if (err.name === 'NotAllowedError') {
          setEnrollError('Passkey registration was cancelled. Please try again or use Email OTP.');
        } else {
          setEnrollError(err.response?.data?.message || err.message || 'Passkey registration failed.');
        }
      } finally {
        setEnrolling(false);
      }
    }, []);
    ```
