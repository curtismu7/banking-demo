  <action>
Create `banking_api_server/services/pingOneUserService.js` with PingOne Management API integration for user CRUD and mayAct attribute management.

**Core functions:**

1. `createPingOneUser(userData)` — POST /v1/environments/{envId}/users
   - Input: `{ email, firstName, lastName, username, phone, address, populationId, role }`
   - Creates user in PingOne with lifecycle status ACCOUNT_OK
   - Assigns to population if populationId provided
   - Returns created user object with id

2. `setUserPassword(userId, password)` — PUT /v1/environments/{envId}/users/{userId}/password
   - Sets password using `application/vnd.pingidentity.password.set+json`
   - forceChange: false for demo convenience

3. `setMayActAttribute(userId, clientId, mode)` — PATCH /v1/environments/{envId}/users/{userId}
   - mode: '1exchange' → mayAct = `{"client_id":"<admin_client_id>"}`
   - mode: '2exchange' → mayAct = `{"client_id":"<ai_agent_client_id>"}`
   - mode: 'clear' → mayAct = null
   - Uses JSON string format as PingOne custom attributes require

4. `getUserProfile(userId)` — GET /v1/environments/{envId}/users/{userId}
   - Returns full user profile including custom attributes
   - Includes mayAct attribute status

5. `updatePingOneUser(userId, updates)` — PATCH /v1/environments/{envId}/users/{userId}
   - Partial update of user profile fields
   - Supports email, name, phone, address

6. `deletePingOneUser(userId)` — DELETE /v1/environments/{envId}/users/{userId}
   - Hard delete from PingOne

7. `listPingOneUsers(filter)` — GET /v1/environments/{envId}/users
   - Optional SCIM filter for searching
   - Returns paginated list

**Implementation pattern:**
```js
const axios = require('axios');
const configStore = require('./configStore');
const { getManagementToken } = require('./pingOneClientService');

function getBaseUrl() {
  const region = configStore.getEffective('pingone_region') || 'com';
  const envId = configStore.getEffective('pingone_environment_id');
  return `https://api.pingone.${region}/v1/environments/${envId}`;
}

async function createPingOneUser(userData) {
  const token = await getManagementToken();
  const baseUrl = getBaseUrl();
  
  const body = {
    email: userData.email,
    username: userData.username || userData.email,
    name: {
      given: userData.firstName,
      family: userData.lastName
    },
    lifecycle: { status: 'ACCOUNT_OK' }
  };
  
  // Optional fields
  if (userData.phone) {
    body.mobilePhone = userData.phone;
    body.primaryPhone = userData.phone;
  }
  if (userData.address) {
    body.address = userData.address;
  }
  if (userData.populationId) {
    body.population = { id: userData.populationId };
  }
  
  const res = await axios.post(`${baseUrl}/users`, body, {
    headers: {
      Authorization: `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    timeout: 15000
  });
  
  return res.data;
}
```

**Error handling:**
- 409 Conflict → user already exists (email/username collision)
- 400 Bad Request → validation error from PingOne
- 401/403 → worker token issue
- Wrap all errors with descriptive messages

Create `banking_api_server/routes/selfServiceUsers.js` with REST API endpoints for self-service user provisioning.

**Endpoints:**

1. `POST /api/self-service/users` — Create new PingOne user
   - Body: `{ email, firstName, lastName, username, password, phone, address, role }`
   - role: 'customer' (default) or 'admin'
   - If role='admin', assigns to admin_population_id
   - After creation, sets password and optionally sets mayAct
   - Returns: created user profile

2. `GET /api/self-service/users/me` — Get current user's PingOne profile
   - Uses req.user.id to fetch from PingOne
   - Returns profile with mayAct status

3. `PATCH /api/self-service/users/me` — Update current user's profile
   - Body: partial update fields (email, name, phone, address)
   - Returns: updated profile

4. `POST /api/self-service/users/mayact` — Configure mayAct on current user
   - Body: `{ enabled: true/false, mode: '1exchange'|'2exchange' }`
   - Sets or clears mayAct attribute
   - Returns: mayAct status

5. `DELETE /api/self-service/users/:userId` — Delete PingOne user (admin only)
   - Requires admin role
   - Returns: confirmation

6. `GET /api/self-service/users/check/:username` — Check if username available
   - Returns: { available: true/false }

**Validation middleware:**
```js
function validateCreateUser(req, res, next) {
  const { email, firstName, lastName, password } = req.body || {};
  const errors = [];
  
  if (!email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) errors.push('Valid email required');
  if (!firstName?.trim()) errors.push('First name required');
  if (!lastName?.trim()) errors.push('Last name required');
  if (!password || password.length < 8) errors.push('Password must be 8+ characters');
  
  if (errors.length > 0) return res.status(400).json({ error: 'validation_error', errors });
  next();
}
```
