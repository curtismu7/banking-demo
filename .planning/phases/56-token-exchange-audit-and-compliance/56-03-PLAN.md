   ```javascript
   /**
    * Validate two-exchange delegation configuration.
    * Called at startup and before first two-exchange attempt.
    * Throws detailed error with remediation steps if validation fails.
    */
   function validateTwoExchangeConfig() {
     const errors = [];
     const warnings = [];
     
     // Require AI Agent credentials
     const aiAgentClientId = configStore.get('PINGONE_AI_AGENT_CLIENT_ID') || process.env.AI_AGENT_CLIENT_ID;
     const aiAgentSecret = configStore.get('PINGONE_AI_AGENT_CLIENT_SECRET') || process.env.AI_AGENT_CLIENT_SECRET;
     if (!aiAgentClientId) errors.push('MISSING: PINGONE_AI_AGENT_CLIENT_ID (AI Agent OAuth app client ID)');
     if (!aiAgentSecret) errors.push('MISSING: PINGONE_AI_AGENT_CLIENT_SECRET (AI Agent OAuth app client secret)');
     
     // Require MCP Exchanger credentials
     const mcpClientId = process.env.AGENT_OAUTH_CLIENT_ID;
     const mcpSecret = process.env.AGENT_OAUTH_CLIENT_SECRET;
     if (!mcpClientId) errors.push('MISSING: AGENT_OAUTH_CLIENT_ID (MCP Token Exchanger OAuth app client ID)');
     if (!mcpSecret) errors.push('MISSING: AGENT_OAUTH_CLIENT_SECRET (MCP Token Exchanger OAuth app client secret)');
     
     // Require explicit audiences (NO hard-coded fallbacks)
     const agentGatewayAud = configStore.getEffective('PINGONE_RESOURCE_AGENT_GATEWAY_URI') || process.env.PINGONE_RESOURCE_AGENT_GATEWAY_URI;
     const mcpGatewayAud = configStore.getEffective('PINGONE_RESOURCE_MCP_GATEWAY_URI') || process.env.PINGONE_RESOURCE_MCP_GATEWAY_URI;
     const intermediateAud = configStore.getEffective('AI_AGENT_INTERMEDIATE_AUDIENCE') || process.env.AI_AGENT_INTERMEDIATE_AUDIENCE;
     const finalAud = configStore.getEffective('PINGONE_RESOURCE_TWO_EXCHANGE_URI') || process.env.PINGONE_RESOURCE_TWO_EXCHANGE_URI;
     
     if (!agentGatewayAud) errors.push('MISSING: PINGONE_RESOURCE_AGENT_GATEWAY_URI (Step 1 audience for AI Agent token)');
     if (!mcpGatewayAud) errors.push('MISSING: PINGONE_RESOURCE_MCP_GATEWAY_URI (Step 3 audience for MCP token)');
     if (!intermediateAud) errors.push('MISSING: AI_AGENT_INTERMEDIATE_AUDIENCE (Step 2 exchange audience, usually MCP server URI)');
     if (!finalAud) errors.push('MISSING: PINGONE_RESOURCE_TWO_EXCHANGE_URI (Step 4 exchange final audience)');
     
     // Audience uniqueness check
     if (intermediateAud && finalAud && intermediateAud === finalAud) {
       warnings.push('WARNING: AI_AGENT_INTERMEDIATE_AUDIENCE equals PINGONE_RESOURCE_TWO_EXCHANGE_URI. They should differ for nested delegation. Verify audience configuration is correct.');
     }
     
     // Throw on errors with remediation steps
     if (errors.length > 0) {
       const errorMsg = [
         'Two-Exchange Delegation Configuration Invalid:',
         ...errors,
         '',
         'Remediation:',
         '1. Set missing environment variables (see ERROR lines above)',
         '2. Or configure via Admin UI → Config page (Two-Exchange section)',
         '3. Verify credentials are from correct PingOne apps:',
         '   - AI_AGENT_CLIENT_ID/SECRET: Super Banking AI Agent App',
         '   - AGENT_OAUTH_CLIENT_ID/SECRET: Super Banking MCP Token Exchanger',
         '4. Verify audience URIs match PingOne token exchange policies',
         '5. Verify ff_two_exchange_delegation is enabled if you intend to use 2-exchange'
       ].join('\n');
       
       const err = new Error(errorMsg);
       err.code = 'TWO_EXCHANGE_CONFIG_INVALID';
       err.httpStatus = 503;
       err.isConfigError = true;
       throw err;
     }
     
     // Log warnings
     if (warnings.length > 0) {
       warnings.forEach(w => logger.warn('[TwoExchange Config] ' + w));
     }
     
     return { valid: true, aiAgentClientId, mcpClientId, audiences: { agentGatewayAud, mcpGatewayAud, intermediateAud, finalAud } };
   }
   ```

3. **Export the function**: Add to module.exports so it can be called from agentMcpTokenService

4. **Document in comments**: Add comment explaining why all values are required (no fallbacks for production)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| BFF → PingOne Exchange | Two exchange requests with different actors (AI Agent, then MCP) |
| Configuration Provider | Environment variables or config store must be trusted |
| Error Messages | May expose configuration details to attackers |

## STRIDE Threat Register

| ID | Category | Component | Threat | Disposition | Mitigation |
|---|----------|-----------|--------|-------------|-----------|
| T-56-03-01 | Information Disclosure | Error Messages | Configuration details exposed in error responses | Mitigate | Log details separately from user-facing messages |
| T-56-03-02 | Denial of Service | Config Validation | Missing config causes repeated validation attempts | Mitigate | Early validation prevents wasted processing |
| T-56-03-03 | Tampering | Configuration | Hard-coded defaults could bypass security policy | Mitigate | Remove all hard-coded fallbacks, require explicit config |
| T-56-03-04 | Spoofing | Nested Act Claims | Incorrect act.act.sub could misidentify actors | Mitigate | Validate both act.sub and act.act.sub structure |

**Mitigations Summary:**
- T-56-03-01: Error messages sanitized before user exposure
- T-56-03-02: Upfront validation prevents repeated failures
- T-56-03-03: All fallbacks removed, explicit config required
- T-56-03-04: Nested act claims validated at each step
</threat_model>
