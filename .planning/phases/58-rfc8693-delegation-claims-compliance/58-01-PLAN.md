# Phase 58-01: RFC 8693 Delegation Claims Compliance Implementation

## Goal

Ensure RFC 8693 token exchange implementation properly follows delegation pattern with correct `may_act` and `act` claim structures, where user tokens contain authorized agent identifiers and exchanged tokens contain complete delegation chains preserving user subject identity.

## Success Criteria

1. **User Token Compliance**: 100% of user tokens contain proper `may_act` claims with authorized agent identifiers
2. **Exchanged Token Structure**: 100% of exchanged tokens preserve user `sub` claim and contain correct nested `act` claims
3. **Delegation Chain Integrity**: Complete delegation chain (user → agent → MCP server) verified in all exchanged tokens
4. **Identity Format Standardization**: All agent and MCP server identifiers use consistent URI format
5. **Error Handling**: Comprehensive validation and error responses for malformed claims

const IDENTIFIER_FORMATS = {
  agent: /^https:\/\/[a-zA-Z0-9.-]+\.pingdemo\.com\/agent\/[a-zA-Z0-9-]+$/,
  mcp_server: /^https:\/\/[a-zA-Z0-9.-]+\.pingdemo\.com\/mcp\/[a-zA-Z0-9-]+$/,

const validateIdentifierFormat = (identifier, type) => {
  const pattern = IDENTIFIER_FORMATS[type];
  if (!pattern.test(identifier)) {
    throw new Error(`Invalid ${type} identifier format: ${identifier}`);
  }
  return true;
};

// Legacy format mapping
const mapLegacyIdentifier = (identifier, type) => {
  const legacyPattern = IDENTIFIER_FORMATS[`legacy_${type}`];
  if (legacyPattern.test(identifier)) {
    // Map legacy format to standardized URI format
    return `https://${type}.pingdemo.com/${type}/${identifier}`;
  }
  return identifier;
};

// Delegation chain validation
const validateDelegationChain = (userToken, exchangedToken) => {
  const chain = reconstructDelegationChain(userToken, exchangedToken);
  
  // Verify chain structure: user -> agent -> mcp_server
  if (chain.length !== 3) {
    throw new Error(`Invalid delegation chain length: ${chain.length}`);
  }
  
  const [user, agent, mcpServer] = chain;
  
  // Verify user subject preservation
  if (user.sub !== exchangedToken.sub) {
    throw new Error('User subject not preserved in exchanged token');
  }
  
  // Verify agent authorization
  if (user.may_act?.sub !== agent.sub) {
    throw new Error('Agent not authorized in may_act claim');
  }
  
  // Verify MCP server identity
  if (exchangedToken.act?.sub !== mcpServer.sub) {
    throw new Error('MCP server identity mismatch in act claim');
  }
  
  return true;
};

// Chain reconstruction
const reconstructDelegationChain = (userToken, exchangedToken) => {
  return [
    { type: 'user', sub: userToken.sub, may_act: userToken.may_act },
    { type: 'agent', sub: userToken.may_act?.sub },
    { type: 'mcp_server', sub: exchangedToken.act?.sub }
  ];
};

// Circular delegation detection
const detectCircularDelegation = (chain) => {
  const subjects = chain.map(node => node.sub);
  const uniqueSubjects = new Set(subjects);
  
  if (subjects.length !== uniqueSubjects.size) {
    throw new Error('Circular delegation detected in chain');
  }
  
  return false;
};
```

// Claim validation middleware
const validateDelegationClaims = (req, res, next) => {
  try {
    const token = req.headers.authorization?.replace('Bearer ', '');
    const claims = decodeJwtClaims(token);
    
    if (!claims) {
      return res.status(401).json({
        error: 'invalid_token',
        message: 'Unable to decode token claims'
      });
    }
    
    // Validate based on token type
    if (claims.may_act) {
      // User token validation
      validateUserTokenClaims(claims);
    } else if (claims.act) {
      // Exchanged token validation
      validateExchangedTokenClaims(claims);
    }
    
    req.tokenClaims = claims;
    next();
  } catch (error) {
    console.error('Claim validation failed:', error);
    return res.status(401).json({
      error: 'claim_validation_failed',
      message: error.message,
      code: 'DELEGATION_CLAIM_INVALID'
    });
  }
};

// User token claim validation
const validateUserTokenClaims = (claims) => {
  if (!claims.sub) {
    throw new Error('Subject claim missing from user token');
  }
  
  if (!claims.may_act || !claims.may_act.sub) {
    throw new Error('may_act claim missing or malformed');
  }
  
  validateIdentifierFormat(claims.may_act.sub, 'agent');
};

// Exchanged token claim validation
const validateExchangedTokenClaims = (claims) => {
  if (!claims.sub) {
    throw new Error('Subject claim missing from exchanged token');
  }
  
  if (!claims.act || !claims.act.sub) {
    throw new Error('act claim missing or malformed');
  }
  
  validateIdentifierFormat(claims.act.sub, 'mcp_server');
  
  if (claims.act.act?.sub) {
    validateIdentifierFormat(claims.act.act.sub, 'agent');
  }
};
```
