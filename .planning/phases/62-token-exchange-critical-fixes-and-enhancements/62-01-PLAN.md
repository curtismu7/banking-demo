```javascript
// Standardized may_act format specification
const MAY_ACT_FORMAT_STANDARD = {
  // URI format: https://agent-domain/{agent-type}/{agent-id}
  uriFormat: 'https://{domain}/{type}/{id}',
  
  // Examples:
  examples: [
    'https://agents.pingdemo.com/ai-agent/banking-assistant-001',
    'https://agents.pingdemo.com/mcp-server/banking-mcp-001',
    'https://agents.pingdemo.com/admin/banking-admin-001'
  ],
  
  // Validation rules:
  validation: {
    scheme: 'https',
    domain: 'agents.pingdemo.com',
    requiredPath: '/{type}/{id}',
    allowedTypes: ['ai-agent', 'mcp-server', 'admin']
  }
};

// may_act format conversion utilities
const MayActFormatConverter = {
  // Convert client_id format to URI format
  clientToUri: (clientId) => {
    if (clientId.startsWith('https://')) return clientId; // Already URI format
    
    const agentType = this.detectAgentType(clientId);
    const agentId = clientId.replace(/^(ai-|mcp-|admin-)/, '');
    return `https://agents.pingdemo.com/${agentType}/${agentId}`;
  },
  
  // Validate URI format
  validateUriFormat: (uri) => {
    const url = new URL(uri);
    if (url.protocol !== 'https:') throw new Error('must use HTTPS');
    if (!url.hostname.includes('agents.pingdemo.com')) throw new Error('invalid domain');
    
    const pathParts = url.pathname.split('/').filter(Boolean);
    if (pathParts.length !== 2) throw new Error('invalid path format');
    
    const [type, id] = pathParts;
    if (!['ai-agent', 'mcp-server', 'admin'].includes(type)) {
      throw new Error('invalid agent type');
    }
    
    return { type, id, valid: true };
  },
  
  // Detect agent type from client_id
  detectAgentType: (clientId) => {
    if (clientId.startsWith('ai-')) return 'ai-agent';
    if (clientId.startsWith('mcp-')) return 'mcp-server';
    if (clientId.startsWith('admin-')) return 'admin';
    throw new Error('cannot detect agent type from client_id');
  }
};
```

```javascript
// RFC 8707 Resource Indicators Implementation
const ResourceIndicators = {
  // Resource definitions
  resources: {
    banking: {
      uri: 'https://banking-api.pingdemo.com',
      scopes: ['banking:read', 'banking:write', 'banking:admin'],
      description: 'Banking API resource server'
    },
    mcp: {
      uri: 'https://mcp-server.pingdemo.com',
      scopes: ['banking:read', 'banking:write', 'banking:agent:invoke'],
      description: 'MCP server resource'
    },
    admin: {
      uri: 'https://admin-api.pingdemo.com',
      scopes: ['admin:read', 'admin:write', 'admin:users'],
      description: 'Admin API resource server'
    }
  },
  
  // Resource indicator validation
  validateResource: (resourceUri, requestedScopes) => {
    const resource = Object.values(this.resources).find(r => r.uri === resourceUri);
    if (!resource) throw new Error(`Unknown resource: ${resourceUri}`);
    
    const invalidScopes = requestedScopes.filter(scope => !resource.scopes.includes(scope));
    if (invalidScopes.length > 0) {
      throw new Error(`Invalid scopes for ${resourceUri}: ${invalidScopes.join(', ')}`);
    }
    
    return { resource, valid: true };
  },
  
  // Enhanced authorization request with resource indicators
  buildAuthRequest: (clientId, redirectUri, scopes, resources) => {
    const params = new URLSearchParams({
      response_type: 'code',
      client_id: clientId,
      redirect_uri: redirectUri,
      scope: scopes.join(' '),
      code_challenge: generateCodeChallenge(),
      code_challenge_method: 'S256'
    });
    
    // Add resource indicators
    if (resources && resources.length > 0) {
      params.append('resource', resources.join(' '));
    }
    
    return `${authEndpoint}?${params.toString()}`;
  },
  
  // Token validation with resource indicators
  validateTokenForResource: (token, resourceUri) => {
    const tokenResources = token.aud || [];
    if (Array.isArray(tokenResources)) {
      if (!tokenResources.includes(resourceUri)) {
        throw new Error(`Token not authorized for resource: ${resourceUri}`);
      }
    } else if (tokenResources !== resourceUri) {
      throw new Error(`Token not authorized for resource: ${resourceUri}`);
    }
    
    return { authorized: true };
  }
};

// Enhanced token exchange with resource indicators
const enhancedTokenExchange = async (subjectToken, actorToken, options) => {
  const {
    audience,
    scopes,
    resource = audience, // Default resource to audience
    actorResource = null // Optional actor resource
  } = options;
  
  // Validate resource indicators
  ResourceIndicators.validateResource(resource, scopes);
  if (actorResource) {
    ResourceIndicators.validateResource(actorResource, ['token:exchange']);
  }
  
  // Build token exchange request with resource indicators
  const exchangeParams = {
    grant_type: 'urn:ietf:params:oauth:grant-type:token-exchange',
    subject_token: subjectToken,
    subject_token_type: 'urn:ietf:params:oauth:token-type:access_token',
    actor_token: actorToken,
    actor_token_type: 'urn:ietf:params:oauth:token-type:access_token',
    scope: scopes.join(' '),
    resource: resource,
    requested_token_type: 'urn:ietf:params:oauth:token-type:access_token'
  };
  
  if (actorResource) {
    exchangeParams.actor_resource = actorResource;
  }
  
  return await performTokenExchange(exchangeParams);
};
```

```javascript
// Simplified Scope Management
const ScopeManager = {
  // Scope definitions and permissions
  scopeDefinitions: {
    'banking:read': {
      description: 'Read banking data',
      resources: ['banking'],
      level: 'read'
    },
    'banking:write': {
      description: 'Write banking data',
      resources: ['banking'],
      level: 'write',
      requires: ['banking:read']
    },
    'banking:admin': {
      description: 'Admin banking operations',
      resources: ['banking', 'admin'],
      level: 'admin',
      requires: ['banking:read', 'banking:write']
    },
    'banking:agent:invoke': {
      description: 'Invoke AI agent operations',
      resources: ['mcp'],
      level: 'agent',
      requires: ['banking:read']
    }
  },
  
  // Simplified scope validation
  validateScopes: (requestedScopes, availableScopes, options = {}) => {
    const { strict = true, allowPartial = false } = options;
    
    // Validate requested scopes exist
    const invalidScopes = requestedScopes.filter(scope => !this.scopeDefinitions[scope]);
    if (invalidScopes.length > 0) {
      throw new Error(`Invalid scopes: ${invalidScopes.join(', ')}`);
    }
    
    // Check scope requirements
    const missingRequirements = this.findMissingRequirements(requestedScopes, availableScopes);
    if (missingRequirements.length > 0) {
      if (strict) {
        throw new Error(`Missing required scopes: ${missingRequirements.join(', ')}`);
      } else if (!allowPartial) {
        throw new Error(`Insufficient scopes: missing ${missingRequirements.join(', ')}`);
      }
    }
    
    // Calculate granted scopes
    const grantedScopes = this.calculateGrantedScopes(requestedScopes, availableScopes);
    
    return {
      requested: requestedScopes,
      available: availableScopes,
      granted: grantedScopes,
      missing: missingRequirements,
      valid: missingRequirements.length === 0 || (allowPartial && grantedScopes.length > 0)
    };
  },
  
  // Find missing required scopes
  findMissingRequirements: (requestedScopes, availableScopes) => {
    const missing = new Set();
    
    for (const scope of requestedScopes) {
      const definition = this.scopeDefinitions[scope];
      if (definition?.requires) {
        for (const required of definition.requires) {
          if (!availableScopes.includes(required)) {
            missing.add(required);
          }
        }
      }
    }
    
    return Array.from(missing);
  },
  
  // Calculate granted scopes (intersection with available)
  calculateGrantedScopes: (requestedScopes, availableScopes) => {
    return requestedScopes.filter(scope => availableScopes.includes(scope));
  },
  
  // Scope narrowing for token exchange
  narrowScopesForExchange: (userScopes, requestedScopes, resource) => {
    // Get resource-specific scopes
    const resourceScopes = this.getResourceScopes(resource);
    
    // Filter user scopes to resource-specific ones
    const userResourceScopes = userScopes.filter(scope => resourceScopes.includes(scope));
    
    // Validate against requested scopes
    const validation = this.validateScopes(requestedScopes, userResourceScopes);
    
    return validation.granted;
  },
  
  // Get scopes available for resource
  getResourceScopes: (resource) => {
    return Object.entries(this.scopeDefinitions)
      .filter(([_, definition]) => definition.resources.includes(resource))
      .map(([scope, _]) => scope);
  }
};

// Simplified token exchange with scope narrowing
const performExchangeWithScopeNarrowing = async (userToken, options) => {
  const { scopes: requestedScopes, resource, audience } = options;
  
  // Extract user token scopes
  const userScopes = extractScopesFromToken(userToken);
  
  // Narrow scopes for the resource
  const narrowedScopes = ScopeManager.narrowScopesForExchange(
    userScopes, 
    requestedScopes, 
    resource
  );
  
  if (narrowedScopes.length === 0) {
    throw new Error('No valid scopes available for requested operation');
  }
  
  // Perform exchange with narrowed scopes
  return await performTokenExchange(userToken, {
    ...options,
    scopes: narrowedScopes,
    audience: audience || resource
  });
};
```

## Migration Guide

### may_act Format Migration

#### Overview
The may_act claim format is being standardized to use URI format instead of client_id format.

#### Migration Steps
1. **Audit Current Usage**: Identify all places using may_act claims
2. **Update Configuration**: Set may_act format preference to 'uri'
3. **Update Code**: Use MayActFormatConverter for format conversion
4. **Test Migration**: Validate all token exchange scenarios
5. **Deploy Changes**: Roll out with backward compatibility
6. **Monitor**: Watch for format-related errors
7. **Remove Old Format**: Deprecate client_id format after migration
