  <name>Task 2: Create routes/rfc9728.js (if not already present)</name>
  <files>banking_api_server/routes/rfc9728.js</files>
  <action>
Create Express router serving RFC 9728 metadata:

```javascript
'use strict';
const express = require('express');
const router = express.Router();
const configStore = require('../services/configStore');

router.get('/', (req, res) => {
  const envId = configStore.getEffective('pingone_environment_id') || process.env.PINGONE_ENVIRONMENT_ID || '';
  const region = configStore.getEffective('pingone_region') || process.env.PINGONE_REGION || 'com';
  const resourceUrl = process.env.PUBLIC_APP_URL || `${req.protocol}://${req.get('host')}`;
  const asUrl = `https://auth.pingone.${region}/${envId}/as`;

  res.set({
    'Content-Type': 'application/json; charset=UTF-8',
    'Cache-Control': 'public, max-age=3600',
    'Access-Control-Allow-Origin': '*',
  });
  res.json({
    resource: resourceUrl,
    authorization_servers: [asUrl],
    bearer_methods_supported: ['header'],
    scopes_supported: ['openid', 'profile', 'email', 'banking:read', 'banking:write', 'banking:admin', 'banking:sensitive', 'ai_agent'],
    resource_documentation: `${resourceUrl}/docs`,
  });
});

module.exports = router;
```
