Create a KV/SQLite migration script so operators running the Admin Config UI can migrate stored config values from old key names to canonical new names without manual re-entry. Update VERCEL_SETUP.md with the complete list of new env vars to add to the Vercel dashboard.

**Purpose:** Operators who saved config through the Admin UI have values stored under old key names (e.g., `agent_oauth_client_id`). The new `getEffective` calls look under `pingone_mcp_token_exchanger_client_id` first — if nothing is in KV under that key, KV storage shows blank even though the value exists under the old key. The migration script fixes this silently.

```js
#!/usr/bin/env node
/**
 * migrate-config-keys.js
 *
 * One-time migration: copy stored config values from old KV/SQLite keys (pre-Phase-88 names)
 * to the new canonical internal key names introduced in commit ab9d1ee + Phase 88.
 *
 * Safe to run multiple times — will not overwrite an already-set new key.
 *
 * Usage:
 *   cd banking_api_server && node scripts/migrate-config-keys.js [--dry-run]
 *
 * Options:
 *   --dry-run   Print what would be migrated without writing anything
 */
'use strict';

const { configStore } = require('../services/configStore');

const MIGRATIONS = [
  // Format: { from: 'old_internal_key', to: 'new_internal_key', description: '...' }
  {
    from: 'agent_oauth_client_id',
    to:   'pingone_mcp_token_exchanger_client_id',
    description: 'MCP Token Exchanger client ID (Super Banking MCP Token Exchanger app)',
  },
  {
    from: 'agent_oauth_client_secret',
    to:   'pingone_mcp_token_exchanger_client_secret',
    description: 'MCP Token Exchanger client secret',
  },
  {
    from: 'agent_oauth_client_scopes',
    to:   'pingone_mcp_token_exchanger_client_scopes',
    description: 'MCP Token Exchanger client scopes',
  },
  {
    from: 'ai_agent_client_id',
    to:   'pingone_ai_agent_client_id',
    description: 'AI Agent client ID (Super Banking AI Agent App)',
  },
  {
    from: 'ai_agent_client_secret',
    to:   'pingone_ai_agent_client_secret',
    description: 'AI Agent client secret',
  },
  {
    from: 'mcp_resource_uri',
    to:   'pingone_resource_mcp_server_uri',
    description: 'MCP Server resource URI (audience)',
  },
  {
    from: 'mcp_gateway_audience',
    to:   'pingone_resource_mcp_gateway_uri',
    description: 'MCP Gateway resource URI (audience)',
  },
  {
    from: 'mcp_resource_uri_two_exchange',
    to:   'pingone_resource_two_exchange_uri',
    description: 'Two-exchange final audience URI',
  },
];

const DRY_RUN = process.argv.includes('--dry-run');

async function main() {
  console.log(`\nPingOne config key migration ${DRY_RUN ? '[DRY RUN]' : ''}`);
  console.log('========================================\n');

  let migrated = 0;
  let skipped = 0;
  let notFound = 0;

  for (const { from, to, description } of MIGRATIONS) {
    const oldValue = await configStore.get(from);
    if (!oldValue) {
      console.log(`  SKIP  ${from} → ${to}  (no stored value under old key)`);
      notFound++;
      continue;
    }

    const existingNew = await configStore.get(to);
    if (existingNew) {
      console.log(`  SKIP  ${from} → ${to}  (new key already has a value)`);
      skipped++;
      continue;
    }

    if (DRY_RUN) {
      const preview = oldValue.length > 12 ? oldValue.slice(0, 8) + '...' : '[short]';
      console.log(`  WOULD MIGRATE  ${from} → ${to}  (${description}, value: ${preview})`);
      migrated++;
    } else {
      await configStore.set(to, oldValue);
      const preview = oldValue.length > 12 ? oldValue.slice(0, 8) + '...' : '[short]';
      console.log(`  MIGRATED  ${from} → ${to}  (value: ${preview})`);
      migrated++;
    }
  }

  console.log(`\nDone. Migrated: ${migrated}, Skipped (new key exists): ${skipped}, Not found: ${notFound}`);
  if (DRY_RUN) {
    console.log('Re-run without --dry-run to apply changes.');
  }
}

main().catch((err) => {
  console.error('Migration failed:', err.message);
  process.exit(1);
});
```
