```javascript
 * Audience (aud) Claim Configuration
 * 
 * Defines audience values for all APIs and services in the system.
 * These values are used to:
 * 1. Validate the 'aud' claim in incoming OAuth tokens (fail-closed)
 * 2. Ensure tokens are intended for the correct recipient (prevent token confusion)

 * BFF API Audience
 * Used to validate user tokens and agent (actor) tokens accessing the banking API

  switch (NODE_ENV) {
    case 'production':
      return process.env.AUD_BFF_API || 'https://banking-api.example.com';
    case 'vercel':
    case 'staging':
      return process.env.AUD_BFF_API || 'https://banking-api.vercel.app';
    default: // development
      return process.env.AUD_BFF_API || 'https://banking-api.local';

  switch (NODE_ENV) {
    case 'production':
      return process.env.AUD_MCP_SERVER || 'mcp.example.com';
    case 'vercel':
    case 'staging':
      return process.env.AUD_MCP_SERVER || 'mcp.pingstaging.com';
    default: // development
      return process.env.AUD_MCP_SERVER || 'mcp.pingdemo.com';

 * PingOne Resource Audiences

  applications: `https://api.pingone.com/v1/environments/${process.env.PINGONE_ENVIRONMENT_ID}/applications`,
  // Add more PingOne resources as needed

  const routeKey = `${method} ${path}`;
  return ROUTE_AUD_MAP[routeKey] || BFF_AUD; // Default to BFF_AUD if not explicitly mapped
```

```javascript
/**
 * Audience (aud) Validation Service
 * 
 * Validates that OAuth tokens include the correct 'aud' (audience) claim,
 * preventing token confusion attacks where a token for one API is used for another.
 * 
 * VALIDATION POLICY: FAIL CLOSED
 * If aud does not match expected value, reject (return invalid, never allow)
 */

const logger = require('../utils/logger');
const audConfig = require('../config/audConfigTemplate');

/**
 * Validate the 'aud' claim in a decoded JWT token
 * 
 * @param {Object} decoded - Decoded JWT token
 * @param {string} expectedAud - Expected audience value (single value or from route config)
 * @returns {{valid: boolean, error?: string, matchedAud?: string}}
 */
function validateAudClaim(decoded, expectedAud) {
  if (!decoded) {
    return { valid: false, error: 'Token not decoded' };
  }

  if (!decoded.aud) {
    logger.warn('Token missing aud claim', { 
      sub: decoded.sub, 
      expectedAud 
    });
    return { valid: false, error: 'Missing aud claim in token' };
  }

  // aud claim can be string or array; normalize
  const tokenAuds = Array.isArray(decoded.aud) ? decoded.aud : [decoded.aud];

  // Check if any of the token's audiences match the expected audience
  const matched = tokenAuds.some(aud => aud === expectedAud);

  if (!matched) {
    logger.warn('Audience mismatch - TOKEN REJECTED (FAIL CLOSED)', {
      tokenAuds,
      expectedAud,
      subject: decoded.sub,
      actor: decoded.act?.sub // if agent token
    });
    return { 
      valid: false, 
      error: `Token aud [${tokenAuds.join(', ')}] does not match expected [${expectedAud}]` 
    };
  }

  logger.debug('Audience validation passed', { 
    aud: decoded.aud, 
    expectedAud 
  });
  return { valid: true, matchedAud: expectedAud };
}

/**
 * Get the expected audience for a given route
 * 
 * @param {string} method - HTTP method
 * @param {string} path - Request path
 * @returns {string} - Expected audience value
 */
function getExpectedAud(method, path) {
  return audConfig.getExpectedAudForRoute(method, path);
}

/**
 * Validate aud claim for a specific route
 * 
 * Combined validation: check that token aud matches route's expected aud
 * 
 * @param {Object} token - Decoded JWT token
 * @param {string} method - HTTP method
 * @param {string} path - Request path
 * @returns {{valid: boolean, error?: string}}
 */
function validateAudForRoute(token, method, path) {
  const expectedAud = getExpectedAud(method, path);
  return validateAudClaim(token, expectedAud);
}

/**
 * Validate aud claim AND additional constraints
 * Used when route needs aud validation + other checks (like scope)
 * 
 * @param {Object} token - Decoded JWT token
 * @param {Object} options - {expectedAud, method, path, requiredScopes, ...}
 * @returns {{valid: boolean, errors: string[]}}
 */
function validateAudAndScopes(token, options) {
  const errors = [];

  // Validate aud
  const audValidation = validateAudClaim(token, options.expectedAud);
  if (!audValidation.valid) {
    errors.push(audValidation.error);
  }

  // Optionally validate scopes
  if (options.requiredScopes && token.scope) {
    const tokenScopes = token.scope.split(' ');
    const hasAllScopes = options.requiredScopes.every(scope => 
      tokenScopes.includes(scope)
    );
    if (!hasAllScopes) {
      errors.push(`Missing required scopes: ${options.requiredScopes.join(', ')}`);
    }
  }

  return {
    valid: errors.length === 0,
    errors
  };
}

/**
 * AudValidationError
 * Thrown when aud validation fails (for explicit error handling)
 */
class AudValidationError extends Error {
  constructor(message, context = {}) {
    super(message);
    this.name = 'AudValidationError';
    this.context = context;
  }
}

module.exports = {
  validateAudClaim,
  getExpectedAud,
  validateAudForRoute,
  validateAudAndScopes,
  AudValidationError,
};
```

```javascript
/**
 * Audience (aud) Validation Middleware
 * 
 * Express middleware that validates the 'aud' claim in incoming OAuth tokens.
 * Implements fail-closed policy: if aud doesn't match, reject with 401.
 * 
 * Validates aud BEFORE the request reaches route handlers.
 * Prevents token confusion attacks (token for API A cannot be used for API B).
 */

const audService = require('../services/audValidationService');
const logger = require('../utils/logger');

/**
 * Middleware: Validate audience (aud) claim in token
 * 
 * Expected usage in server.js:
 *   app.use(audValidationMiddleware);
 * 
 * For custom routes with specific aud requirements:
 *   router.post('/restricted', audValidationMiddleware, restrictedRoute);
 * 
 * To skip aud validation on specific routes:
 *   const skipAudValidation = (req, res, next) => {
 *     req.skipAudValidation = true;
 *     next();
 *   };
 *   app.use('/public', skipAudValidation, publicRoutes);
 */
function audValidationMiddleware(req, res, next) {
  // Allow certain routes to skip aud validation (e.g., health checks, public endpoints)
  if (req.skipAudValidation || 
      req.path === '/health' || 
      req.path === '/metrics' ||
      req.path === '/api/public' ||
      req.path.startsWith('/.well-known')) {
    return next();
  }

  // If no token, skip (let other middleware handle auth)
  if (!req.user || !req.user.decoded) {
    return next();
  }

  // Validate aud for this request
  const validation = audService.validateAudForRoute(
    req.user.decoded,
    req.method,
    req.path
  );

  if (!validation.valid) {
    logger.warn('Aud validation failed - rejecting request', {
      path: req.path,
      method: req.method,
      subject: req.user.decoded.sub,
      tokenAud: req.user.decoded.aud,
      expectedAud: audService.getExpectedAud(req.method, req.path),
      error: validation.error
    });

    return res.status(401).json({
      error: 'invalid_token',
      error_description: 'Token audience does not match this API',
      details: validation.error
    });
  }

  // Attach expected aud to request for route handlers to use (optional)
  req.expectedAud = audService.getExpectedAud(req.method, req.path);

  logger.debug('Aud validation passed', {
    path: req.path,
    aud: req.user.decoded.aud
  });

  next();
}

module.exports = audValidationMiddleware;
```
