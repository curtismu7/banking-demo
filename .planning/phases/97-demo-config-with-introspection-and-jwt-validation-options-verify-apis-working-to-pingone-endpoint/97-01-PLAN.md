```javascript
/**
 * Token Validation Mode Configuration
 * 
 * Supports two validation strategies:
 * 1. introspection — Call PingOne introspection endpoint (RFC 7662) for real-time token validation
 *    Pros: Detects revoked tokens, real-time authorization, most secure
 *    Cons: Adds latency (mitigated by caching), requires network access to PingOne
 * 
 * 2. jwt — Validate JWT signature locally (RS256 with PingOne public key)
 *    Pros: Fast, no network call needed, works offline
 *    Cons: Cannot detect revoked tokens, delayed authorization changes
 */

const VALIDATION_MODES = {
  INTROSPECTION: 'introspection',
  JWT: 'jwt'
};

const SUPPORTED_MODES = Object.values(VALIDATION_MODES);

// Default to introspection (RFC 7662 is more secure)
const DEFAULT_MODE = VALIDATION_MODES.INTROSPECTION;

let currentMode = process.env.VALIDATION_MODE || DEFAULT_MODE;

/**
 * Validate that mode is supported
 */
function isValidMode(mode) {
  return SUPPORTED_MODES.includes(mode);
}

/**
 * Get current validation mode
 * @returns {string} 'introspection' or 'jwt'
 */
function getValidationMode() {
  return currentMode;
}

/**
 * Set validation mode (runtime change)
 * Note: Changes mode for current session only; does not persist to .env
 * @param {string} mode - 'introspection' or 'jwt'
 * @throws {Error} If mode is not supported
 */
function setValidationMode(mode) {
  if (!isValidMode(mode)) {
    throw new Error(`Invalid validation mode: ${mode}. Supported: ${SUPPORTED_MODES.join(', ')}`);
  }
  currentMode = mode;
}

/**
 * Get description of validation mode
 */
function getModeDescription(mode = currentMode) {
  switch (mode) {
    case VALIDATION_MODES.INTROSPECTION:
      return 'Introspection (Real-time, RFC 7662) — Validates tokens via PingOne, detects revoked tokens';
    case VALIDATION_MODES.JWT:
      return 'JWT Local Validation (Fast, Offline) — Validates signature locally, cannot detect revocation';
    default:
      return 'Unknown mode';
  }
}

module.exports = {
  VALIDATION_MODES,
  SUPPORTED_MODES,
  DEFAULT_MODE,
  getValidationMode,
  setValidationMode,
  isValidMode,
  getModeDescription
};
```

/**
 * Health Check Routes
 * 
 * GET /api/health/introspection — Test PingOne introspection endpoint connectivity
 */

const express = require('express');
const router = express.Router();
const tokenIntrospectionService = require('../services/tokenIntrospectionService');
const logger = require('../utils/logger');

/**
 * GET /api/health/introspection
 * 
 * Tests whether the BFF can reach the PingOne introspection endpoint.
 * This is useful for operators to verify connectivity before demo operations.
 * 
 * Response structure:
 * {
 *   status: 'connected' | 'failed',
 *   endpoint: 'https://auth.pingone.com/...',
 *   timestamp: '2026-04-08T...',
 *   details: {
 *     responseTime: 45,  // ms
 *     cached: false,     // was result from cache?
 *     error?: 'error message if failed'
 *   }
 * }
 */
router.get('/introspection', async (req, res) => {
  try {
    const startTime = Date.now();
    
    // Test token — use a test/demo token that will be declined but proves endpoint reachable
    // Use a minimal token structure that won't cause security issues
    const testToken = process.env.TEST_TOKEN_FOR_HEALTH_CHECK || 'test_token_' + Date.now();
    
    // Call introspection service
    // Note: testToken will likely be invalid, but that's OK — we're testing connectivity, not token validity
    const result = await tokenIntrospectionService.validateToken(testToken, {
      skipValidityCheck: true  // Only test connectivity, don't enforce token validity
    });
    
    const responseTime = Date.now() - startTime;
    
    logger.info('Introspection health check passed', {
      endpoint: process.env.PINGONE_INTROSPECTION_ENDPOINT,
      responseTime
    });
    
    return res.status(200).json({
      status: 'connected',
      endpoint: process.env.PINGONE_INTROSPECTION_ENDPOINT || 'Not configured',
      timestamp: new Date().toISOString(),
      details: {
        responseTime,
        cached: false,
        message: 'PingOne introspection endpoint is reachable'
      }
    });
  } catch (error) {
    logger.warn('Introspection health check failed', {
      endpoint: process.env.PINGONE_INTROSPECTION_ENDPOINT,
      error: error.message
    });
    
    return res.status(503).json({
      status: 'failed',
      endpoint: process.env.PINGONE_INTROSPECTION_ENDPOINT || 'Not configured',
      timestamp: new Date().toISOString(),
      details: {
        error: error.message,
        hint: 'Check PINGONE_INTROSPECTION_ENDPOINT env var and network connectivity'
      }
    });
  }
});

module.exports = router;
```

```typescript
/**
 * ConfigTokenValidation Component
 * 
 * UI for configuring token validation mode (introspection vs jwt).
 * Displays current mode, allows toggling, and provides health check button.
 */

import React, { useState, useEffect } from 'react';
import styles from './ConfigTokenValidation.module.css';

interface ValidationModeConfig {
  mode: 'introspection' | 'jwt';
  description: string;
}

interface HealthCheckResult {
  status: 'connected' | 'failed';
  endpoint: string;
  timestamp: string;
  details: {
    responseTime?: number;
    error?: string;
    message?: string;
  };
}

export const ConfigTokenValidation: React.FC = () => {
  const [validationMode, setValidationMode] = useState<'introspection' | 'jwt'>('introspection');
  const [healthStatus, setHealthStatus] = useState<HealthCheckResult | null>(null);
  const [isTestingConnection, setIsTestingConnection] = useState(false);
  const [testError, setTestError] = useState<string | null>(null);

  // Load current validation mode on mount
  useEffect(() => {
    const loadValidationMode = async () => {
      try {
        const response = await fetch('/api/config/validation-mode');
        if (response.ok) {
          const data = await response.json();
          setValidationMode(data.mode);
        }
      } catch (error) {
        console.error('Failed to load validation mode:', error);
      }
    };
    
    loadValidationMode();
  }, []);

  // Handle mode change
  const handleModeChange = async (newMode: 'introspection' | 'jwt') => {
    try {
      const response = await fetch('/api/config/validation-mode', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ mode: newMode })
      });
      
      if (response.ok) {
        setValidationMode(newMode);
        setTestError(null);
      } else {
        setTestError('Failed to update validation mode');
      }
    } catch (error) {
      setTestError('Error updating validation mode: ' + (error as Error).message);
    }
  };

  // Test PingOne connectivity
  const handleHealthCheck = async () => {
    setIsTestingConnection(true);
    setTestError(null);
    
    try {
      const response = await fetch('/api/health/introspection');
      const data = await response.json() as HealthCheckResult;
      setHealthStatus(data);
      
      if (response.status !== 200) {
        setTestError(data.details?.error || 'Health check failed');
      }
    } catch (error) {
      setTestError('Failed to test connection: ' + (error as Error).message);
    } finally {
      setIsTestingConnection(false);
    }
  };

  const modeDescriptions = {
    introspection: 'Real-time validation via PingOne (RFC 7662) — Detects revoked tokens, adds ~50ms latency (cached)',
    jwt: 'Local JWT signature validation (Fast, Offline) — Cannot detect revoked tokens'
  };

  return (
    <div className={styles.configTokenValidation}>
      <h3>Token Validation Mode</h3>
      
      <div className={styles.modeSelector}>
        <label>
          <input
            type="radio"
            name="validation-mode"
            value="introspection"
            checked={validationMode === 'introspection'}
            onChange={() => handleModeChange('introspection')}
          />
          <strong>Introspection (Recommended)</strong>
          <p className={styles.description}>{modeDescriptions.introspection}</p>
        </label>
        
        <label>
          <input
            type="radio"
            name="validation-mode"
            value="jwt"
            checked={validationMode === 'jwt'}
            onChange={() => handleModeChange('jwt')}
          />
          <strong>JWT Local Validation</strong>
          <p className={styles.description}>{modeDescriptions.jwt}</p>
        </label>
      </div>

      <div className={styles.healthCheck}>
        <button
          onClick={handleHealthCheck}
          disabled={isTestingConnection}
          className={styles.testButton}
        >
          {isTestingConnection ? 'Testing...' : 'Test PingOne Connection'}
        </button>

        {healthStatus && (
          <div className={`${styles.healthResult} ${styles[healthStatus.status]}`}>
            <strong>{healthStatus.status === 'connected' ? '✓ Connected' : '✗ Failed'}</strong>
            <p>Endpoint: {healthStatus.endpoint}</p>
            {healthStatus.details.responseTime && (
              <p>Response time: {healthStatus.details.responseTime}ms</p>
            )}
            {healthStatus.details.message && <p>{healthStatus.details.message}</p>}
            {healthStatus.details.error && <p className={styles.error}>{healthStatus.details.error}</p>}
            <p className={styles.timestamp}>Tested: {new Date(healthStatus.timestamp).toLocaleTimeString()}</p>
          </div>
        )}

        {testError && (
          <div className={styles.errorMessage}>
            <strong>Error:</strong> {testError}
          </div>
        )}
      </div>

      <div className={styles.info}>
        <h4>About Token Validation</h4>
        <ul>
          <li><strong>Introspection:</strong> Recommended for production. Calls PingOne's introspection endpoint to validate tokens in real-time.</li>
          <li><strong>JWT:</strong> Used as fallback when introspection is not available. Validates token signature locally but cannot detect revoked tokens.</li>
          <li><strong>Fallback:</strong> If introspection fails, the system automatically falls back to JWT validation.</li>
        </ul>
      </div>
    </div>
  );
};

export default ConfigTokenValidation;
```

```markdown
# Token Validation: Introspection vs JWT

This guide explains how the banking demo validates OAuth tokens and helps you choose the right validation strategy for your use case.

## Overview

The banking demo supports two token validation approaches:

| Strategy | How It Works | Real-time Revocation | Speed | Offline | Best For |
|----------|-------------|-------------------|-------|---------|-----------|
| **Introspection (RFC 7662)** | Calls PingOne endpoint to validate token | Yes ✓ | ~50ms (cached) | No | User-facing APIs, high security |
| **JWT Local Validation** | Validates RSA signature locally | No ✗ | ~1ms | Yes ✓ | Internal APIs, fallback mode |

## 1. Introspection (Recommended for Production)

### How It Works

1. **User makes API request** with OAuth token
2. **BFF calls PingOne introspection endpoint** (RFC 7662)
3. **PingOne returns** token status (active, expired, revoked, etc.)
4. **BFF caches result** for 30 seconds (configurable)
5. **Request allowed/denied** based on introspection result

### When to Use

✓ User-facing APIs (transactions, transfers, sensitive data access)  
✓ Compliance requirements (PCI, SOX, HIPAA) requiring real-time validation  
✓ High-security operations (money transfer, account deletion)  
✓ Any operation where token revocation must be detected immediately  

### Configuration

Set environment variables:

```bash
# PingOne Introspection Endpoint
PINGONE_INTROSPECTION_ENDPOINT=https://auth.pingone.com/environments/{env-id}/oauth2/introspect

# Worker credentials (must have introspection scope)
WORKER_CLIENT_ID=your_worker_app_id
WORKER_CLIENT_SECRET=your_worker_app_secret

# Optional: Cache TTL (default 30 seconds)
INTROSPECTION_CACHE_TTL=30

# Optional: Fallback to JWT if introspection fails
INTROSPECTION_FALLBACK_TO_JWT=true
```

### Testing Connectivity

Use the demo config UI:

1. Navigate to **Config** > **Token Validation Mode**
2. Click **"Test PingOne Connection"** button
3. View result: Connected (✓) or Failed (✗)
4. If failed, check:
   - Network connectivity to PingOne
   - WORKER_CLIENT_ID and WORKER_CLIENT_SECRET are correct
   - PINGONE_INTROSPECTION_ENDPOINT is set and correct
   - Worker app has "Token Introspection" scope enabled

### Troubleshooting

**503 Service Unavailable**
- PingOne introspection endpoint is not reachable
- Check network access, firewall rules, DNS

**401 Unauthorized**
- Worker credentials are invalid
- Verify WORKER_CLIENT_ID and WORKER_CLIENT_SECRET in env

**400 Bad Request**
- Token format is invalid
- Ensure token is a valid Bearer token (JWT or opaque string)

**Slow Response (>1000ms)**
- Network latency to PingOne is high
- Introspection caching is not working (check logs)
- Consider increasing cache TTL if acceptable for your use case

## 2. JWT Local Validation (Fallback)

### How It Works

1. **User makes API request** with OAuth token
2. **BFF validates token signature locally** (RSA RS256)
3. **BFF decodes JWT claims** (exp, iat, aud, scope, etc.)
4. **Request allowed/denied** based on validation checks

### When to Use

✓ Internal APIs where token revocation latency is acceptable  
✓ Fallback when PingOne is temporarily unavailable  
✓ High-throughput systems where introspection adds unacceptable latency  
✓ Offline-capable systems (edge compute, embedded APIs)  

### Configuration

```bash
# Use JWT for initial validation
VALIDATION_MODE=jwt

# Still have PingOne credentials available for fallback/upgrade
PINGONE_INTROSPECTION_ENDPOINT=...
WORKER_CLIENT_ID=...
WORKER_CLIENT_SECRET=...
```

### Limitations

⚠ **Cannot detect revoked tokens** — If a user's token is revoked via PingOne, the JWT will still be accepted for ~1 hour (until expiry)  
⚠ **Delayed authorization** — Changes to user permissions (scope, attributes) take ~1 hour to take effect  
⚠ **Not recommended for** — High-security operations involving money, PII access, or policy-protected transactions  

## 3. Automatic Fallback

The BFF implements graceful fallback:

1. **Preferential:** Try introspection first
2. **On failure:** Fall back to JWT validation
3. **Log both:** Log which method was used for every request

This means:
- If PingOne is down, the system stays operational (degraded mode)
- Users can continue accessing APIs with JWT validation
- Once PingOne is recovered, introspection resumes automatically

## 6. Performance Considerations

### Introspection Latency

- **Without cache:** ~100-200ms per token validation
- **With cache (30s TTL):** ~1-2ms for cached tokens, ~100-200ms for cache misses
- **Impact on demo:** Negligible for demo purposes, but notable in high-throughput production

### Cache Hit Rate

- First login → Cache miss (~200ms)
- Subsequent requests from same user → Cache hit (~1ms) until TTL expires
- Typical hit rate → 70-80% in active demo sessions

### Optimization Tips

- Increase `INTROSPECTION_CACHE_TTL` to 60-120 seconds if revocation latency is acceptable
- Reduce `INTROSPECTION_CACHE_TTL` to 10 seconds if immediate revocation is required
- Monitor response times in `/api/introspect` logs to detect bottlenecks

## 7. Security Implications

### Attack Vector: Token Reuse

- **Introspection prevents:** ✓ Revoked token detection stops reuse
- **JWT allows:** ✗ Revoked tokens still accepted until expiry

Mitigation: Use short token expiry (15 min) + introspection for user-facing APIs.

### Attack Vector: Stolen Tokens

- **Introspection:** Detects if token is being used from unusual location (via Authorize rules)
- **JWT:** No way to detect or stop usage

Mitigation: Use introspection + device binding (future phase).

### PingOne Outage Scenario

- **Introspection mode:** Falls back to JWT, degraded security but operational
- **JWT mode:** Unaffected, maintains full security
- **Recommended:** Have both modes configured, with preferences per route
